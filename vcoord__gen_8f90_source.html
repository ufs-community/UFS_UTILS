5 !
6 !! This subprogram generates hybrid coordinate interface profiles
7 !! from a few given parameters. The hybrid coordinate is intended to start
8 !! out at the bottom in pure sigma and end up at the top in pure pressure,
9 !! with a smooth transition in between. The pressure thickness is close to
10 !! quadratic in pressure, with maximum thicknesses in the middle of the domain.
11 !! The coordinate pressure will have continuous second derivatives in level.
12 !!
13 !! The hybrid coordinate is returned in terms of vectors AK and BK, where
14 !! the interface pressure is defined as A+B*ps where ps is surface pressure
15 !! Thus A=0 in regions of pure sigma and B=0 in regions of pure sigma.
16 !! At the bottom, A(0)=0 and B(0)=1 so that surface pressure is the bottom
17 !! boundary condition, while at the top, A(levs)=ptop and B(levs)=0 so that
18 !! the constant top pressure (which can be zero) is the top boundary condition.
19 !!
20 !! The procedure for the calculation is described in the remarks section below.
21 !!
22 !! Program history log:
23 !! - 2008-08-01 Mark Iredell
24 !!
25 !! @param[in] levs integer number of levels
26 !! @param[in] lupp integer number of levels below pupp
27 !! @param[in] pbot real nominal surface pressure (Pa)
28 !! @param[in] psig real nominal pressure where coordinate changes
29 !! from pure sigma (Pa)
30 !! @param[in] ppre real nominal pressure where coordinate changes
31 !! to pure pressure (Pa)
32 !! @param[in] pupp real nominal pressure where coordinate changes
33 !! to upper atmospheric profile (Pa)
34 !! @param[in] ptop real pressure at top (Pa)
35 !! @param[in] dpbot real coordinate thickness at bottom (Pa)
36 !! @param[in] dpsig real thickness of zone within which coordinate changes
37 !! to pure sigma (Pa)
38 !! @param[in] dppre real thickness of zone within which coordinate changes
39 !! to pure pressure (Pa)
40 !! @param[in] dpupp real coordinate thickness at pupp (Pa)
41 !! @param[in] dptop real coordinate thickness at top (Pa)
42 !! @param[out] pmin real minimum surface pressure (Pa)
43 !! @param[out] ak real(0:levs) a coordinate values, bottom to top (Pa)
44 !! @param[out] bk real(0:levs) b coordinate values, bottom to top ()
45 !!
46 !! Subprograms called:
47 !! - ludcmp lower and upper triangular decomposition
48 !! - lubksb lower and upper triangular back substitution
49 !!
50 !! <pre>
51 !! Example: Create the operational GFS 64-level hybrid coordinate.
52 !! real(8) pmin,ak(0:64),bk(0:64)
53 !! call vcoord_gen(64,64,100000.,99400.,7000.,0.,0.,500.,1200.,18000.,60.,60.,&
54 !! pmin,ak,bk)
55 !! print '(2i6)',2,64
56 !! print '(f12.3,f12.8)',(ak(k),bk(k),k=0,64)
57 !! end
58 !!
59 !! Graphical description of parameters and zones:
60 !! ptop --- ----- ----------------------
61 !! ... dptop
62 !! --- zone U (upper atmos)
63 !! ...
64 !! pupp --- ----- ----------------------
65 !! ... dpupp
66 !! --- -----
67 !! ... zone P (pure pressure)
68 !! ---
69 !! ...
70 !! ppre --- ----- ----------------------
71 !! ...
72 !! --- dppre zone T1 (transition 1)
73 !! ...
74 !! --- ----- ----------------------
75 !! ...
76 !! ---
77 !! ... zone T2 (transition 2)
78 !! ---
79 !! ...
80 !! --- ----- ----------------------
81 !! ...
82 !! --- dpsig zone T3 (transition 3)
83 !! ...
84 !! psig --- ----- ----------------------
85 !! ...
86 !! --- ----- zone S (pure sigma)
87 !! ... dpbot
88 !! pbot --- ----- ----------------------
89 !! </pre>
90 !!
91 !! Detailed procedure description:
92 !! 1 STEP 1.
93 !! The pressure profile is computed with respect to the given reference
94 !! surface pressure pbot. For this surface pressure, the 'sigma' thicknesses
95 !! dsig are assumed to be proportional to a quadratic polynomial in sigma sig
96 !! with zero intercepts sig1 and sig2 somewhere below and above the model
97 !! domain, respectively. That is,
98 !! dsig ~ (sig2-sig)*(sig-sig1)*dk
99 !! Integrating this differential equation gives
100 !! sig = (sig1*exp(c1*k+c2)+sig2)/(exp(c1*k+c2)+1)
101 !! The required boundary conditions sig(0)=1 and sig(levs)=0
102 !! fix the proportionality and integration constants c1 and c2.
103 !! The two crossing parameters (sig1 and sig2) are determined
104 !! by two input sigma thickness conditions dsig/dk at the bottom and top
105 !! which are respectively given as dpbot/(pbot-pupp) and dpupp/(pbot-pupp).
106 !! The crossing parameters are computed using Newton-Raphson iteration.
107 !! This procedure fixes the pressure profile for surface pressure pbot.
108 !! 2 STEP 2.
109 !! The pressure profile is computed with respect to a minimum surface pressure.
110 !! This minimum surface pressure pmin is yet to be determined.
111 !! Divide the profile into zones:
112 !! zone U (pure pressure) from pupp to ptop
113 !! zone P (pure pressure) from pupp to ppre
114 !! zone T1 (transition 1) from ppre to ppre+dppre
115 !! zone T2 (transition 2) from ppre+dppre to psig-dpsig
116 !! zone T3 (transition 3) from psig-dpsig to psig
117 !! zone S (pure "sigma") from psig to pmin
118 !! (here sigma=p/ps so that d(ln(p))/dk is horizontally uniform)
119 !! The pressure profile in the pure pressure zone P is set from step 1.
120 !! The pressure thicknesses in zone T1 is set to be quadratic in level k.
121 !! The pressure thicknesses in zone T2 is set to be linear in level k.
122 !! The pressure thicknesses in zone T3 is set to be quadratic in level k.
123 !! The pressure profile in the pure sigma zone S is also set from step 1.
124 !! Thus there are nine unknowns:
125 !! the 3 polynomial coefficients in zone T1
126 !! the 2 polynomial coefficients in zone T2
127 !! the 3 polynomial coefficients in zone T3
128 !! and the 1 minimum surface pressure.
129 !! The nine conditions to determine these unknowns are:
130 !! the thickness and its derivative match at zone P and T1 boundary
131 !! the thickness and its derivative match at zone T1 and T2 boundary
132 !! the thickness and its derivative match at zone T2 and T3 boundary
133 !! the thickness and its derivative match at zone T3 and S boundary
134 !! the sum of the thicknesses of zones T1, T2, T3, and S is pmin-ppre
135 !! The unknowns are computed using standard linear decomposition.
136 !! This procedure fixes the pressure profile for surface pressure pmin.
137 !! 3 STEP 3.
138 !! (Step 3 skipped if lupp=levs, in which case pupp=ptop and dpupp=dptop.)
139 !! The pressure in zone U is assumed to be the exponential of a cubic
140 !! polynomial in level k. The function must match the pressure at pupp,
141 !! as well as the thickness and its derivative there, and the pressure
142 !! at ptop+dptop at the second to top level. The latter 3 conditions
143 !! are determined by using standard linear decomposition.
144 !! 4 STEP 4.
145 !! For an arbitrary surface pressure, the pressure profile is an linear
146 !! combination of the pressure profiles for surface pressures pbot and pmin
147 !! <pre>
148 !! p(psfc)=p(pbot)*(psfc-pmin)/(pbot-pmin)+p(pmin)*(pbot-psfc)/(pbot-pmin)
149 !! </pre>
150 !! from which the hybrid coordinate profiles ak and bk are found such that
151 !! <pre>
152 !! p(psfc)=ak+bk*psfc
153 !! </pre>
154 !!
155 subroutine vcoord_gen(levs,lupp,pbot,psig,ppre,pupp,ptop,&
156  dpbot,dpsig,dppre,dpupp,dptop,pmin,ak,bk)
157  implicit none
158  integer,intent(in):: levs,lupp
159  real,intent(in):: pbot,psig,ppre,pupp,ptop
160  real,intent(in):: dpbot,dpsig,dppre,dpupp,dptop
161  real,intent(out):: pmin,ak(0:levs),bk(0:levs)
162  integer,parameter:: lo=100,li=10 ! outer and inner N-R iterations
163  real pdif ! thickness from pbot to pupp
164  real delb ! delta sig at bot
165  real delt ! delta sig at top
166  real sig1 ! crossing parameter 1
167  real sig2 ! crossing parameter 2
168  real c1 ! proportionality constant
169  real c2 ! integration constant
170  real sig ! sig variable
171  real dsig ! delta sig variable
172  real delbio0 ! initial guess at delta sig at bot
173  real deltio0 ! initial guess at delta sig at top
174  real delbio ! guess at delta sig at bot
175  real deltio ! guess at delta sig at top
176  real c1sig1 ! d(c1)/d(sig1)
177  real c1sig2 ! d(c1)/d(sig2)
178  real p(2) ! rhs in N-R iteration
179  real fjac(2,2) ! lhs in N-R iteration
180  integer indx(2) ! permutations in N-R iteration
181  real ppred ! pressure at T1-T2 boundary
182  real spre ! sig at P-T1 boundary
183  real spred ! sig at T1-T2 boundary
184  real rkpre ! level at P-T1 boundary
185  real rkpred ! level at T1-T2 boundary
186  real pkpre ! dp/dk at P-T1 boundary
187  real pkkpre ! d2p/dk2 at P-T1 boundary
188  real psigd ! pressure at T2-T3 boundary
189  real ssig ! sig at T3-S boundary
190  real ssigd ! sig at T2-T3 boundary
191  real rksig ! level at T3-S boundary
192  real rksigd ! level at T2-T3 boundary
193  real pksig ! dp/dk at T3-S boundary
194  real pkksig ! d2p/dk2 at T3-S boundary
195  real p2sig ! pressure at T3-S boundary for pmin surface pressure
196  real p2sigd ! pressure at T2-T3 boundary for pmin surface pressure
197  real p2pred ! pressure at T1-T2 boundary for pmin surface pressure
198  real x2(9) ! rhs in linear solver
199  real a2(9,9) ! lhs in linear solver
200  integer indx2(9) ! permutations in linear solver
201  real pkupp ! dp/dk at U-P boundary
202  real pkkupp ! d2p/dk2 at U-P boundary
203  real x3(3) ! rhs in linear solver
204  real a3(3,3) ! lhs in linear solver
205  integer indx3(3) ! permutations in linear solver
206  real p1 ! pressure variable for pbot surface pressure
207  real p2 ! pressure variable for pmin surface pressure
208  real d ! determinant permutation
209  integer io,ii,k
210 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
211 ! STEP 1.
212  pdif=pbot-pupp
213  delb=dpbot/pdif
214  delt=dpupp/pdif
215  sig1=1+delb
216  sig2=-delt
217  c1=log(-sig2*(1-sig1)/sig1/(sig2-1))/lupp
218  c2=log((sig2-1)/(1-sig1))
219  sig=1
220  dsig=(sig2-sig)*(sig-sig1)*c1/(sig1-sig2)
221  delbio0=-dsig
222  sig=0
223  dsig=(sig2-sig)*(sig-sig1)*c1/(sig1-sig2)
224  deltio0=-dsig
225 ! Newton-Raphson iterations
226  do io=1,lo
227  delbio=delbio0+(delb-delbio0)*io/lo
228  deltio=deltio0+(delt-deltio0)*io/lo
229  do ii=1,li
230  c1sig1=-1/(sig1*(1-sig1)*lupp)
231  c1sig2=-1/(sig2*(sig2-1)*lupp)
232  sig=1
233  dsig=(sig2-sig)*(sig-sig1)*c1/(sig1-sig2)
234  p(1)=-delbio-dsig
235  fjac(1,1)=((-c1*(sig+sig2)+(sig-sig1)*c1sig1*(sig1+sig2)) &
236  *(sig2-sig)/(sig1+sig2)**2)
237  fjac(1,2)=((c1*(sig+sig1)+(sig2-sig)*c1sig2*(sig1+sig2)) &
238  *(sig-sig1)/(sig1+sig2)**2)
239  sig=0
240  dsig=(sig2-sig)*(sig-sig1)*c1/(sig1-sig2)
241  p(2)=-deltio-dsig
242  fjac(2,1)=((-c1*(sig+sig2)+(sig-sig1)*c1sig1*(sig1+sig2)) &
243  *(sig2-sig)/(sig1+sig2)**2)
244  fjac(2,2)=((c1*(sig+sig1)+(sig2-sig)*c1sig2*(sig1+sig2)) &
245  *(sig-sig1)/(sig1+sig2)**2)
246  call ludcmp(fjac,2,2,indx,d)
247  call lubksb(fjac,2,2,indx,p)
248  sig1=sig1+p(1)
249  sig2=sig2+p(2)
250  c1=log(-sig2*(1-sig1)/sig1/(sig2-1))/lupp
251  c2=log((sig2-1)/(1-sig1))
252  enddo
253  enddo
254 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
255 ! STEP 2.
256 ! Compute minimum surface pressure
257  ppred=ppre+dppre
258  spre=(ppre-pupp)/pdif
259  spred=(ppred-pupp)/pdif
260  rkpre=(log((spre-sig2)/(sig1-spre))-c2)/c1
261  rkpred=(log((spred-sig2)/(sig1-spred))-c2)/c1
262  pkpre=pdif*c1*(sig2-spre)*(spre-sig1)/(sig1-sig2)
263  pkkpre=pkpre*c1*(sig2+sig1-2*spre)/(sig1-sig2)
264  psigd=psig-dpsig
265  ssig=(psig-pupp)/pdif
266  ssigd=(psigd-pupp)/pdif
267  rksig=(log((ssig-sig2)/(sig1-ssig))-c2)/c1
268  rksigd=(log((ssigd-sig2)/(sig1-ssigd))-c2)/c1
269  pksig=pdif*c1*(sig2-ssig)*(ssig-sig1)/(sig1-sig2)
270  pkksig=pksig*c1*(sig2+sig1-2*ssig)/(sig1-sig2)
271  x2=0
272  a2=0
273  x2(1)=pkpre
274  a2(1,1)=1
275  a2(1,2)=rkpre
276  a2(1,3)=rkpre**2
277  x2(2)=pkkpre
278  a2(2,2)=1
279  a2(2,3)=2*rkpre
280  a2(3,1)=1
281  a2(3,2)=rkpred
282  a2(3,3)=rkpred**2
283  a2(3,4)=-1
284  a2(3,5)=-rkpred
285  a2(4,2)=1
286  a2(4,3)=2*rkpred
287  a2(4,5)=-1
288  a2(5,4)=-1
289  a2(5,5)=-rksigd
290  a2(5,6)=1
291  a2(5,7)=rksigd
292  a2(5,8)=rksigd**2
293  a2(6,5)=-1
294  a2(6,7)=1
295  a2(6,8)=2*rksigd
296  a2(7,6)=1
297  a2(7,7)=rksig
298  a2(7,8)=rksig**2
299  a2(7,9)=-pksig/pbot
300  a2(8,7)=1
301  a2(8,8)=2*rksig
302  a2(8,9)=-pkksig/pbot
303  x2(9)=ppre
304  a2(9,1)=(rkpre-rkpred)
305  a2(9,2)=(rkpre**2-rkpred**2)/2
306  a2(9,3)=(rkpre**3-rkpred**3)/3
307  a2(9,4)=(rkpred-rksigd)
308  a2(9,5)=(rkpred**2-rksigd**2)/2
309  a2(9,6)=(rksigd-rksig)
310  a2(9,7)=(rksigd**2-rksig**2)/2
311  a2(9,8)=(rksigd**3-rksig**3)/3
312  a2(9,9)=psig/pbot
313  call ludcmp(a2,9,9,indx2,d)
314  call lubksb(a2,9,9,indx2,x2)
315  pmin=x2(9)
316  p2sig=psig/pbot*pmin
317  p2sigd=p2sig &
318  +x2(6)*(rksigd-rksig) &
319  +x2(7)*(rksigd**2-rksig**2)/2 &
320  +x2(8)*(rksigd**3-rksig**3)/3
321  p2pred=p2sigd &
322  +x2(4)*(rkpred-rksigd) &
323  +x2(5)*(rkpred**2-rksigd**2)/2
324 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
325 ! STEP 3.
326  if(lupp.lt.levs) then
327  pkupp=pdif*c1*(sig2-0)*(0-sig1)/(sig1-sig2)
328  pkkupp=pkupp*c1*(sig2+sig1-2*0)/(sig1-sig2)
329  x3=0
330  a3=0
331  x3(1)=pkupp
332  a3(1,1)=pupp
333  x3(2)=pkkupp*pupp-pkupp**2
334  a3(2,2)=pupp**2
335  x3(3)=log((ptop+dptop)/pupp)
336  a3(3,1)=levs-1-lupp
337  a3(3,2)=(levs-1-lupp)**2/2
338  a3(3,3)=(levs-1-lupp)**3/3
339  call ludcmp(a3,3,3,indx3,d)
340  call lubksb(a3,3,3,indx3,x3)
341  endif
342 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
343 ! STEP 4.
344 ! Compute hybrid interface values
345  ak(0)=0
346  bk(0)=1
347  do k=1,levs-1
348  if(k.ge.lupp) then
349  p1=pupp*exp(x3(1)*(k-lupp)+x3(2)*(k-lupp)**2/2+x3(3)*(k-lupp)**3/3)
350  else
351  p1=(sig1*exp(c1*k+c2)+sig2)/(exp(c1*k+c2)+1)*pdif+pupp
352  endif
353  if(k.ge.rkpre) then
354  p2=p1
355  elseif(k.ge.rkpred) then
356  p2=p2pred+x2(1)*(k-rkpred) &
357  +x2(2)*(k**2-rkpred**2)/2 &
358  +x2(3)*(k**3-rkpred**3)/3
359  elseif(k.ge.rksigd) then
360  p2=p2sigd+x2(4)*(k-rksigd) &
361  +x2(5)*(k**2-rksigd**2)/2
362  elseif(k.ge.rksig) then
363  p2=p2sig+x2(6)*(k-rksig) &
364  +x2(7)*(k**2-rksig**2)/2 &
365  +x2(8)*(k**3-rksig**3)/3
366  else
367  p2=p1/pbot*pmin
368  endif
369  ak(k)=(p2*pbot-p1*pmin)/(pbot-pmin)
370  bk(k)=(p1-p2)/(pbot-pmin)
371  enddo
372  ak(levs)=ptop
373  bk(levs)=0
374 end subroutine
