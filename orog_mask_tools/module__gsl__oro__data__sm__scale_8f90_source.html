368 !$OMP PRIVATE(I_BLK,J_BLK,S_II,S_JJ,E_II,E_JJ,LON_BLK,LAT_BLK,II_M,JJ_M) &

390  end do

431  end if

434  ! Check for "shaved" block due to proximity to north pole
435  if ( (e_jj(j_blk).eq.-999).and.(s_jj(j_blk).ne.-999) ) then
436  e_jj(j_blk) = dimy_fine ! northern boundary of shaved block
437  ! Reassign latitude of block center

439  end if

469  end do

474  end do

509  !
510  ! Calculate standard deviation of subgrid-scale terrain height

512 
513  ! Calculate mean height
514  sum2 = 0._real_kind
515  nfinepoints = ii_m*jj_m
516  do jj = 1,jj_m
517  do ii = 1,ii_m

519  end do
520  end do

522 
523  ! Calculate standard deviation
524  sum2 = 0._real_kind
525  do jj = 1,jj_m
526  do ii = 1,ii_m
527  sum2 = sum2 + ( zs(ii,jj) - zs_mean )**2
528  end do
529  end do
530  std_dev(cell_count) = sqrt( sum2/real(nfinepoints,real_kind) )
531 
532  !
533  ! Calculate convexity of sub-grid-scale terrain
534  !
535 
536  sum2 = 0._real_kind
537  sum4 = 0._real_kind
538  do jj = 1,jj_m
539  do ii = 1,ii_m
540  sum2 = sum2 + ( zs(ii,jj) - zs_mean )**2
541  sum4 = sum4 + ( zs(ii,jj) - zs_mean )**4
542  end do
543  end do
544 
545  var = sum2 / real(nfinepoints,real_kind)
546  if ( abs(var) < 1.0e-05_real_kind ) then
547  convexity(cell_count) = 0._real_kind
548  else
549  convexity(cell_count) = min( sum4 / ( var**2 * &
550  real(nfinepoints,real_kind) ), max_convexity )
551  end if

553 
554  !
555  ! Calculate orographic asymmetries
556  !
557 
558  ! OA1 -- orographic asymmetry in West direction
559  nu = 0
560  nd = 0
561  do jj = 1,jj_m
562  if(mod(ii_m,2).eq.0.) then
563  do ii = 1,ii_m/2 ! left half of box
564  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
565  end do
566  else
567  do ii = 1,ii_m/2+1 ! left half of box
568  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
569  end do
570  endif
571  do ii = ii_m/2 + 1, ii_m ! right half of box
572  if ( zs(ii,jj) > zs_mean ) nd = nd + 1
573  end do
574  end do
575  if ( nu + nd > 0 ) then
576  oa1(cell_count) = real((nu - nd),real_kind) / &
577  real((nu + nd),real_kind)
578  else
579  oa1(cell_count) = 0._real_kind
580  end if
581 
582  ! OA2 -- orographic asymmetry in South direction
583  nu = 0
584  nd = 0
585  if(mod(jj_m,2).eq.0.) then
586  do jj = 1,jj_m/2 ! bottom half of box
587  do ii = 1,ii_m
588  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
589  end do
590  end do
591  else
592  do jj = 1,jj_m/2+1 ! bottom half of box
593  do ii = 1,ii_m
594  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
595  end do
596  end do
597  endif
598  do jj = jj_m/2 + 1,jj_m ! top half of box
599  do ii = 1, ii_m
600  if ( zs(ii,jj) > zs_mean ) nd = nd + 1
601  end do
602  end do
603  if ( nu + nd > 0 ) then
604  oa2(cell_count) = real((nu - nd),real_kind) / &
605  real((nu + nd),real_kind)
606  else
607  oa2(cell_count) = 0._real_kind
608  end if
609 
610  ! OA3 -- orographic asymmetry in South-West direction
611  nu = 0
612  nd = 0
613  ratio = real(jj_m,real_kind)/real(ii_m,real_kind)
614  do jj = 1,jj_m
615  do ii = 1,ii_m
616  if ( nint(real(ii,real_kind)*ratio) <= (jj_m - jj + 1) ) then
617  ! south-west half of box
618  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
619  endif
620  if ( nint(real(ii,real_kind)*ratio) >= (jj_m - jj + 1) ) then
621  ! north-east half of box
622  if ( zs(ii,jj) > zs_mean ) nd = nd + 1
623  end if
624  end do
625  end do
626  if ( nu + nd > 0 ) then
627  oa3(cell_count) = real((nu - nd),real_kind) / &
628  real((nu + nd),real_kind)
629  else
630  oa3(cell_count) = 0._real_kind
631  end if
632 
633  ! OA4 -- orographic asymmetry in North-West direction
634  nu = 0
635  nd = 0
636  ratio = real(jj_m,real_kind)/real(ii_m,real_kind)
637  do jj = 1,jj_m
638  do ii = 1,ii_m
639  if ( nint(real(ii,real_kind)*ratio) <= jj ) then
640  ! north-west half of box
641  if ( zs(ii,jj) > zs_mean ) nu = nu + 1
642  end if
643  if ( nint(real(ii,real_kind)*ratio) >= jj ) then
644  ! south-east half of box
645  if ( zs(ii,jj) > zs_mean ) nd = nd + 1
646  end if
647  end do
648  end do
649  if ( nu + nd > 0 ) then
650  oa4(cell_count) = real((nu - nd),real_kind) / &
651  real((nu + nd),real_kind)
652  else
653  oa4(cell_count) = 0._real_kind
654  end if

656 
657  !
658  ! Calculate orographic effective lengths
659  !
660 
661  ! OL1 -- orographic effective length for Westerly flow
662  nw = 0
663  nt = 0
664  do jj = max(jj_m/4,1), 3*jj_m/4
665  ! within central east-west band of box
666  do ii = 1, ii_m
667  if ( zs(ii,jj) > zs_mean ) nw = nw + 1
668  nt = nt + 1
669  end do
670  end do
671  if ( nt /= 0 ) then
672  ol1(cell_count) = real(nw,real_kind) / real(nt,real_kind)
673  else
674  ol1(cell_count) = 0._real_kind
675  end if
676 
677  ! OL2 -- orographic effective length for Southerly flow
678  nw = 0
679  nt = 0
680  do jj = 1, jj_m
681  do ii = max(ii_m/4,1), 3*ii_m/4
682  ! within central north-south band of box
683  if ( zs(ii,jj) > zs_mean ) nw = nw + 1
684  nt = nt + 1
685  end do
686  end do
687  if ( nt /= 0 ) then
688  ol2(cell_count) = real(nw,real_kind) / real(nt,real_kind)
689  else
690  ol2(cell_count) = 0._real_kind
691  end if
692 
693  ! OL3 -- orographic effective length for South-Westerly flow
694  nw = 0
695  nt = 0
696  do jj = 1, jj_m/2
697  do ii = 1, ii_m/2
698  if ( zs(ii,jj) > zs_mean ) nw = nw + 1
699  nt = nt + 1
700  end do
701  end do
702  do jj = jj_m/2+1, jj_m
703  do ii = ii_m/2+1, ii_m
704  if ( zs(ii,jj) > zs_mean ) nw = nw + 1
705  nt = nt + 1
706  end do
707  end do
708  if ( nt /= 0 ) then
709  ol3(cell_count) = real(nw,real_kind) / real(nt,real_kind)
710  else
711  ol3(cell_count) = 0._real_kind
712  end if
713 
714  ! OL4 -- orographic effective length for North-Westerly flow
715  nw = 0
716  nt = 0
717  do jj = jj_m/2+1, jj_m
718  do ii = 1, ii_m/2
719  if ( zs(ii,jj) > zs_mean ) nw = nw + 1
720  nt = nt + 1
721  end do
722  end do
723  do jj = 1, jj_m/2
724  do ii = ii_m/2+1, ii_m
725  if ( zs(ii,jj) > zs_mean ) nw = nw + 1

728  end do

731  else
732  ol4(cell_count) = 0._real_kind
733  end if
