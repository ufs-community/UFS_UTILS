593 /* distance between cartesian grid on the earth */
594 double cartesian_dist(double x1, double y1, double x2, double y2, double f_plane_latitude)
595 {
596  double dist = 0.0;
597 
598  if(x1 == x2)
599  dist = fabs(y2-y1)*D2R*RADIUS;
600  else if(y1 == y2)
601  dist = fabs(x2-x1)*D2R*RADIUS*cos(f_plane_latitude*D2R);
602  else
603  mpp_error("create_lonlat_grid: This is not rectangular grid");
604 
605  return dist;
606 }
