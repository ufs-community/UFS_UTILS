348  SUBROUTINE newps(IM,ZS,PS,IMX,KM,P,T,Q,ZSNEW,PSNEW)

350 !
351 ! SUBPROGRAM: NEWPS COMPUTE NEW SURFACE PRESSURE
352 ! PRGMMR: IREDELL ORG: W/NMC23 DATE: 92-10-31
353 !
354 ! ABSTRACT: COMPUTES A NEW SURFACE PRESSURE GIVEN A NEW OROGRAPHY.
355 ! THE NEW PRESSURE IS COMPUTED ASSUMING A HYDROSTATIC BALANCE
356 ! AND A CONSTANT TEMPERATURE LAPSE RATE. BELOW GROUND, THE
357 ! LAPSE RATE IS ASSUMED TO BE -6.5 K/KM.
358 !
359 ! PROGRAM HISTORY LOG:
360 ! 91-10-31 MARK IREDELL
361 !
362 ! USAGE: CALL NEWPS(IM,ZS,PS,IMX,KM,P,T,Q,ZSNEW,PSNEW)
363 ! INPUT ARGUMENT LIST:
364 ! IM INTEGER NUMBER OF POINTS TO COMPUTE
365 ! ZS REAL (IM) OLD OROGRAPHY (M)
366 ! PS REAL (IM) OLD SURFACE PRESSURE (PA)
367 ! IMX INTEGER FIRST DIMENSION
368 ! KM INTEGER NUMBER OF LEVELS
369 ! P REAL (IMX,KM) PRESSURES (PA)
370 ! T REAL (IMX,KM) TEMPERATURES (K)
371 ! Q REAL (IMX,KM) SPECIFIC HUMIDITIES (KG/KG)
372 ! ZSNEW REAL (IM) NEW OROGRAPHY (M)
373 ! OUTPUT ARGUMENT LIST:
374 ! PSNEW REAL (IM) NEW SURFACE PRESSURE (PA)
375 !
376 ! ATTRIBUTES:
377 ! LANGUAGE: FORTRAN
378 !
379 !C$$$
380  INTEGER,INTENT(IN)::im,imx,km
381  REAL,INTENT(IN):: zs(im),ps(im),p(imx,km)
382  REAL,INTENT(IN):: t(imx,km),q(imx,km)
383  REAL,INTENT(IN):: zsnew(im)
384  REAL,INTENT(OUT):: psnew(im)
385  parameter(beta=-6.5e-3,epsilon=1.e-9)
386  parameter(g=9.80665,rd=287.05,rv=461.50)
387  parameter(gor=g/rd,fv=rv/rd-1.)
388  REAL zu(im)
389  ftv(at,aq)=at*(1+fv*aq)
390  fgam(apu,atvu,apd,atvd)=-gor*log(atvd/atvu)/log(apd/apu)
391  fz0(ap,atv,azd,apd)=azd+atv/gor*log(apd/ap)
392  fz1(ap,atv,azd,apd,agam)=azd-atv/agam*((apd/ap)**(-agam/gor)-1)
393  fp0(az,azu,apu,atvu)=apu*exp(-gor/atvu*(az-azu))
394  fp1(az,azu,apu,atvu,agam)=apu*(1+agam/atvu*(az-azu))**(-gor/agam)
395 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
396 ! COMPUTE SURFACE PRESSURE BELOW THE ORIGINAL GROUND
397  ls=0
398  k=1
399  gamma=beta

401 ! if (zsnew(i) == zs(i)) then
402 ! psnew(i) = ps(i)
403 ! else
404  pu=p(i,k)
405  tvu=ftv(t(i,k),q(i,k))
406  zu(i)=fz1(pu,tvu,zs(i),ps(i),gamma)
407  IF(zsnew(i).LE.zu(i)) THEN
408  pu=p(i,k)
409  tvu=ftv(t(i,k),q(i,k))
410  IF(abs(gamma).GT.epsilon) THEN
411  psnew(i)=fp1(zsnew(i),zu(i),pu,tvu,gamma)
412  ELSE
413  psnew(i)=fp0(zsnew(i),zu(i),pu,tvu)
414  ENDIF
415  ELSE
416  psnew(i)=0
417  ls=ls+1
418  ENDIF
419 ! endif
420  ENDDO
421 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
422 ! COMPUTE SURFACE PRESSURE ABOVE THE ORIGINAL GROUND
423  DO k=2,km
424  IF(ls.GT.0) THEN
425  DO i=1,im
426  IF(psnew(i).EQ.0) THEN
427  pu=p(i,k)
428  tvu=ftv(t(i,k),q(i,k))
429  pd=p(i,k-1)
430  tvd=ftv(t(i,k-1),q(i,k-1))
431  gamma=fgam(pu,tvu,pd,tvd)
432  IF(abs(gamma).GT.epsilon) THEN
433  zu(i)=fz1(pu,tvu,zu(i),pd,gamma)
434  ELSE
435  zu(i)=fz0(pu,tvu,zu(i),pd)
436  ENDIF
437  IF(zsnew(i).LE.zu(i)) THEN
438  IF(abs(gamma).GT.epsilon) THEN
439  psnew(i)=fp1(zsnew(i),zu(i),pu,tvu,gamma)
440  ELSE
441  psnew(i)=fp0(zsnew(i),zu(i),pu,tvu)
442  ENDIF
443  ls=ls-1
444  ENDIF
445  ENDIF
446  ENDDO
447  ENDIF
448  ENDDO
449 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
450 ! COMPUTE SURFACE PRESSURE OVER THE TOP
451  IF(ls.GT.0) THEN
452  k=km
453  gamma=0
454  DO i=1,im
455  IF(psnew(i).EQ.0) THEN
456  pu=p(i,k)
457  tvu=ftv(t(i,k),q(i,k))
458  psnew(i)=fp0(zsnew(i),zu(i),pu,tvu)
459  ENDIF
460  ENDDO
461  ENDIF
462  END SUBROUTINE newps

473 ! THE INTERPOLATION IS CUBIC LAGRANGIAN IN LOG PRESSURE
474 ! WITH A MONOTONIC CONSTRAINT IN THE CENTER OF THE DOMAIN.
475 ! IN THE OUTER INTERVALS IT IS LINEAR IN LOG PRESSURE.
476 ! OUTSIDE THE DOMAIN, FIELDS ARE GENERALLY HELD CONSTANT,
477 ! EXCEPT FOR TEMPERATURE AND HUMIDITY BELOW THE INPUT DOMAIN,
478 ! WHERE THE TEMPERATURE LAPSE RATE IS HELD FIXED AT -6.5 K/KM AND
479 ! THE RELATIVE HUMIDITY IS HELD CONSTANT.
480 !
481 ! PROGRAM HISTORY LOG:
482 ! 91-10-31 MARK IREDELL
483 !
484 ! USAGE: CALL VINTG(IM,IX,KM1,KM2,NT,P1,U1,V1,T1,Q1,P2,
485 ! & U2,V2,T2,Q2)
486 ! INPUT ARGUMENT LIST:
487 ! IM INTEGER NUMBER OF POINTS TO COMPUTE
488 ! IX INTEGER FIRST DIMENSION
489 ! KM1 INTEGER NUMBER OF INPUT LEVELS
490 ! KM2 INTEGER NUMBER OF OUTPUT LEVELS
491 ! NT INTEGER NUMBER OF TRACERS
492 ! P1 REAL (IX,KM1) INPUT PRESSURES
493 ! ORDERED FROM BOTTOM TO TOP OF ATMOSPHERE
494 ! U1 REAL (IX,KM1) INPUT ZONAL WIND
495 ! V1 REAL (IX,KM1) INPUT MERIDIONAL WIND
496 ! T1 REAL (IX,KM1) INPUT TEMPERATURE (K)
497 ! Q1 REAL (IX,KM1,NT) INPUT TRACERS (HUMIDITY FIRST)
498 ! P2 REAL (IX,KM2) OUTPUT PRESSURES
499 ! OUTPUT ARGUMENT LIST:
500 ! U2 REAL (IX,KM2) OUTPUT ZONAL WIND
501 ! V2 REAL (IX,KM2) OUTPUT MERIDIONAL WIND
502 ! T2 REAL (IX,KM2) OUTPUT TEMPERATURE (K)
503 ! Q2 REAL (IX,KM2,NT) OUTPUT TRACERS (HUMIDITY FIRST)
504 ! DTDP2 REAL (IX,KM2) OUTPUT DTDP
505 !
506 ! SUBPROGRAMS CALLED:
507 ! TERP3 CUBICALLY INTERPOLATE IN ONE DIMENSION
508 !
509 ! ATTRIBUTES:
510 ! LANGUAGE: FORTRAN
511 !
512 !C$$$
513  REAL p1(ix,km1),u1(ix,km1),v1(ix,km1),t1(ix,km1),q1(ix,km1,nt) &
514  ,w1(ix,km1)
515  REAL p2(ix,km2),u2(ix,km2),v2(ix,km2),t2(ix,km2),q2(ix,km2,nt) &
516  ,w2(ix,km2)
517 ! REAL,optional :: DTDP2(IX,KM2)
518  parameter(dltdz=-6.5e-3*287.05/9.80665)
519  parameter(dlpvdrt=-2.5e6/461.50)
520 
521  REAL,allocatable :: z1(:,:),z2(:,:)
522  REAL,allocatable :: c1(:,:,:),c2(:,:,:),j2(:,:,:)
523 
524  allocate (z1(im+1,km1),z2(im+1,km2))
525  allocate (c1(im+1,km1,4+nt),c2(im+1,km2,4+nt),j2(im+1,km2,4+nt))
526 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
527 ! COMPUTE LOG PRESSURE INTERPOLATING COORDINATE
528 ! AND COPY INPUT WIND, TEMPERATURE, HUMIDITY AND OTHER TRACERS
529 !$OMP PARALLEL DO DEFAULT(SHARED), &
530 !$OMP& PRIVATE(K,I)
531  DO k=1,km1
532  DO i=1,im
533  z1(i,k) = -log(p1(i,k))
534  c1(i,k,1) = u1(i,k)
535  c1(i,k,2) = v1(i,k)
536  c1(i,k,3) = w1(i,k)
537  c1(i,k,4) = t1(i,k)
538  c1(i,k,5) = q1(i,k,1)
539  ENDDO
540  ENDDO
541 !$OMP END PARALLEL DO
542  DO n=2,nt
543  DO k=1,km1
544  DO i=1,im
545  c1(i,k,4+n) = q1(i,k,n)
546  ENDDO
547  ENDDO
548  ENDDO
549 !$OMP PARALLEL DO DEFAULT(SHARED), &
550 !$OMP& PRIVATE(K,I)
551  DO k=1,km2
552  DO i=1,im
553  z2(i,k) = -log(p2(i,k))
554  ENDDO
555  ENDDO
556 !$OMP END PARALLEL DO
557 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
558 ! PERFORM LAGRANGIAN ONE-DIMENSIONAL INTERPOLATION
559 ! THAT IS 4TH-ORDER IN INTERIOR, 2ND-ORDER IN OUTSIDE INTERVALS
560 ! AND 1ST-ORDER FOR EXTRAPOLATION.
561  CALL terp3(im,1,1,1,1,4+nt,(im+1)*km1,(im+1)*km2, &
562  km1,im+1,im+1,z1,c1,km2,im+1,im+1,z2,c2,j2)
563 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
564 ! COPY OUTPUT WIND, TEMPERATURE, HUMIDITY AND OTHER TRACERS
565 ! EXCEPT BELOW THE INPUT DOMAIN, LET TEMPERATURE INCREASE WITH A FIXED
566 ! LAPSE RATE AND LET THE RELATIVE HUMIDITY REMAIN CONSTANT.
567  DO k=1,km2
568  DO i=1,im
569  u2(i,k)=c2(i,k,1)
570  v2(i,k)=c2(i,k,2)
571  w2(i,k)=c2(i,k,3)
572  dz=z2(i,k)-z1(i,1)
573  IF(dz.GE.0) THEN
574  t2(i,k)=c2(i,k,4)
575  q2(i,k,1)=c2(i,k,5)
576 !jaa DTDP2(I,K)=-J2(I,K,4)/P2(I,K)
577  ELSE
578  t2(i,k)=t1(i,1)*exp(dltdz*dz)
579  q2(i,k,1)=q1(i,1,1)*exp(dlpvdrt*(1/t2(i,k)-1/t1(i,1))-dz)
580 !jaa DTDP2(I,K)=-T2(I,K)*DLTDZ/P2(I,K)
581  ENDIF
582  ENDDO
583  ENDDO
584  DO n=2,nt
585  DO k=1,km2
586  DO i=1,im
587  q2(i,k,n)=c2(i,k,4+n)
588  ENDDO
589  ENDDO
590  ENDDO
591  DEALLOCATE (z1,z2,c1,c2,j2)
592  END SUBROUTINE vintg

594  SUBROUTINE terp3(IM,IXZ1,IXQ1,IXZ2,IXQ2,NM,NXQ1,NXQ2, &
595  & km1,kxz1,kxq1,z1,q1,km2,kxz2,kxq2,z2,q2,j2)
596 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
597 !
598 ! SUBPROGRAM: TERP3 CUBICALLY INTERPOLATE IN ONE DIMENSION
599 ! PRGMMR: IREDELL ORG: W/NMC23 DATE: 98-05-01
600 !
601 ! ABSTRACT: INTERPOLATE FIELD(S) IN ONE DIMENSION ALONG THE COLUMN(S).
602 ! THE INTERPOLATION IS CUBIC LAGRANGIAN WITH A MONOTONIC CONSTRAINT
603 ! IN THE CENTER OF THE DOMAIN. IN THE OUTER INTERVALS IT IS LINEAR.
604 ! OUTSIDE THE DOMAIN, FIELDS ARE HELD CONSTANT.
605 !
606 ! PROGRAM HISTORY LOG:
607 ! 98-05-01 MARK IREDELL
608 ! 1999-01-04 IREDELL USE ESSL SEARCH
609 !
610 ! USAGE: CALL TERP3(IM,IXZ1,IXQ1,IXZ2,IXQ2,NM,NXQ1,NXQ2,
611 ! & KM1,KXZ1,KXQ1,Z1,Q1,KM2,KXZ2,KXQ2,Z2,Q2,J2)
612 ! INPUT ARGUMENT LIST:
613 ! IM INTEGER NUMBER OF COLUMNS
614 ! IXZ1 INTEGER COLUMN SKIP NUMBER FOR Z1
615 ! IXQ1 INTEGER COLUMN SKIP NUMBER FOR Q1
616 ! IXZ2 INTEGER COLUMN SKIP NUMBER FOR Z2
617 ! IXQ2 INTEGER COLUMN SKIP NUMBER FOR Q2
618 ! NM INTEGER NUMBER OF FIELDS PER COLUMN
619 ! NXQ1 INTEGER FIELD SKIP NUMBER FOR Q1
620 ! NXQ2 INTEGER FIELD SKIP NUMBER FOR Q2
621 ! KM1 INTEGER NUMBER OF INPUT POINTS
622 ! KXZ1 INTEGER POINT SKIP NUMBER FOR Z1
623 ! KXQ1 INTEGER POINT SKIP NUMBER FOR Q1
624 ! Z1 REAL (1+(IM-1)*IXZ1+(KM1-1)*KXZ1)
625 ! INPUT COORDINATE VALUES IN WHICH TO INTERPOLATE
626 ! (Z1 MUST BE STRICTLY MONOTONIC IN EITHER DIRECTION)
627 ! Q1 REAL (1+(IM-1)*IXQ1+(KM1-1)*KXQ1+(NM-1)*NXQ1)
628 ! INPUT FIELDS TO INTERPOLATE
629 ! KM2 INTEGER NUMBER OF OUTPUT POINTS
630 ! KXZ2 INTEGER POINT SKIP NUMBER FOR Z2
631 ! KXQ2 INTEGER POINT SKIP NUMBER FOR Q2
632 ! Z2 REAL (1+(IM-1)*IXZ2+(KM2-1)*KXZ2)
633 ! OUTPUT COORDINATE VALUES TO WHICH TO INTERPOLATE
634 ! (Z2 NEED NOT BE MONOTONIC)
635 !
636 ! OUTPUT ARGUMENT LIST:
637 ! Q2 REAL (1+(IM-1)*IXQ2+(KM2-1)*KXQ2+(NM-1)*NXQ2)
638 ! OUTPUT INTERPOLATED FIELDS
639 ! J2 REAL (1+(IM-1)*IXQ2+(KM2-1)*KXQ2+(NM-1)*NXQ2)
640 ! OUTPUT INTERPOLATED FIELDS CHANGE WRT Z2
641 !
642 ! SUBPROGRAMS CALLED:
643 ! RSEARCH SEARCH FOR A SURROUNDING REAL INTERVAL
644 !
645 ! ATTRIBUTES:
646 ! LANGUAGE: FORTRAN
647 !
648 !C$$$
649  IMPLICIT NONE
650  INTEGER im,ixz1,ixq1,ixz2,ixq2,nm,nxq1,nxq2
651  INTEGER km1,kxz1,kxq1,km2,kxz2,kxq2
652  INTEGER i,k1,k2,n
653  REAL z1(1+(im-1)*ixz1+(km1-1)*kxz1)
654  REAL q1(1+(im-1)*ixq1+(km1-1)*kxq1+(nm-1)*nxq1)
655  REAL z2(1+(im-1)*ixz2+(km2-1)*kxz2)
656  REAL q2(1+(im-1)*ixq2+(km2-1)*kxq2+(nm-1)*nxq2)
657  REAL j2(1+(im-1)*ixq2+(km2-1)*kxq2+(nm-1)*nxq2)
658  REAL ffa(im),ffb(im),ffc(im),ffd(im)
659  REAL gga(im),ggb(im),ggc(im),ggd(im)
660  INTEGER k1s(im,km2)
661  REAL z1a,z1b,z1c,z1d,q1a,q1b,q1c,q1d,z2s,q2s,j2s
662 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
663 ! FIND THE SURROUNDING INPUT INTERVAL FOR EACH OUTPUT POINT.
664  CALL rsearch(im,km1,ixz1,kxz1,z1,km2,ixz2,kxz2,z2,1,im,k1s)
665 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
666 ! GENERALLY INTERPOLATE CUBICALLY WITH MONOTONIC CONSTRAINT
667 ! FROM TWO NEAREST INPUT POINTS ON EITHER SIDE OF THE OUTPUT POINT,
668 ! BUT WITHIN THE TWO EDGE INTERVALS INTERPOLATE LINEARLY.
669 ! KEEP THE OUTPUT FIELDS CONSTANT OUTSIDE THE INPUT DOMAIN.
670 
671 !$OMP PARALLEL DO DEFAULT(PRIVATE) SHARED(IM,IXZ1,IXQ1,IXZ2) , &
672 !$OMP& SHARED(IXQ2,NM,NXQ1,NXQ2,KM1,KXZ1,KXQ1,Z1,Q1,KM2,KXZ2) , &
673 !$OMP& SHARED(KXQ2,Z2,Q2,J2,K1S)
674 
675  DO k2=1,km2
676  DO i=1,im
677  k1=k1s(i,k2)
678  IF(k1.EQ.1.OR.k1.EQ.km1-1) THEN
679  z2s=z2(1+(i-1)*ixz2+(k2-1)*kxz2)
680  z1a=z1(1+(i-1)*ixz1+(k1-1)*kxz1)
681  z1b=z1(1+(i-1)*ixz1+(k1+0)*kxz1)
682  ffa(i)=(z2s-z1b)/(z1a-z1b)
683  ffb(i)=(z2s-z1a)/(z1b-z1a)
684  gga(i)=1/(z1a-z1b)
685  ggb(i)=1/(z1b-z1a)
686  ELSEIF(k1.GT.1.AND.k1.LT.km1-1) THEN
687  z2s=z2(1+(i-1)*ixz2+(k2-1)*kxz2)
688  z1a=z1(1+(i-1)*ixz1+(k1-2)*kxz1)
689  z1b=z1(1+(i-1)*ixz1+(k1-1)*kxz1)
690  z1c=z1(1+(i-1)*ixz1+(k1+0)*kxz1)
691  z1d=z1(1+(i-1)*ixz1+(k1+1)*kxz1)
692  ffa(i)=(z2s-z1b)/(z1a-z1b)* &
693  (z2s-z1c)/(z1a-z1c)* &
694  (z2s-z1d)/(z1a-z1d)
695  ffb(i)=(z2s-z1a)/(z1b-z1a)* &
696  (z2s-z1c)/(z1b-z1c)* &
697  (z2s-z1d)/(z1b-z1d)
698  ffc(i)=(z2s-z1a)/(z1c-z1a)* &
699  (z2s-z1b)/(z1c-z1b)* &
700  (z2s-z1d)/(z1c-z1d)
701  ffd(i)=(z2s-z1a)/(z1d-z1a)* &
702  (z2s-z1b)/(z1d-z1b)* &
703  (z2s-z1c)/(z1d-z1c)
704  gga(i)= 1/(z1a-z1b)* &
705  (z2s-z1c)/(z1a-z1c)* &
706  (z2s-z1d)/(z1a-z1d)+ &
707  (z2s-z1b)/(z1a-z1b)* &
708  1/(z1a-z1c)* &
709  (z2s-z1d)/(z1a-z1d)+ &
710  (z2s-z1b)/(z1a-z1b)* &
711  (z2s-z1c)/(z1a-z1c)* &
712  1/(z1a-z1d)
713  ggb(i)= 1/(z1b-z1a)* &
714  (z2s-z1c)/(z1b-z1c)* &
715  (z2s-z1d)/(z1b-z1d)+ &
716  (z2s-z1a)/(z1b-z1a)* &
717  1/(z1b-z1c)* &
718  (z2s-z1d)/(z1b-z1d)+ &
719  (z2s-z1a)/(z1b-z1a)* &
720  (z2s-z1c)/(z1b-z1c)* &
721  1/(z1b-z1d)
722  ggc(i)= 1/(z1c-z1a)* &
723  (z2s-z1b)/(z1c-z1b)* &
724  (z2s-z1d)/(z1c-z1d)+ &
725  (z2s-z1a)/(z1c-z1a)* &
726  1/(z1c-z1b)* &
727  (z2s-z1d)/(z1c-z1d)+ &
728  (z2s-z1a)/(z1c-z1a)* &
729  (z2s-z1b)/(z1c-z1b)* &
730  1/(z1c-z1d)
731  ggd(i)= 1/(z1d-z1a)* &
732  (z2s-z1b)/(z1d-z1b)* &
733  (z2s-z1c)/(z1d-z1c)+ &
734  (z2s-z1a)/(z1d-z1a)* &
735  1/(z1d-z1b)* &
736  (z2s-z1c)/(z1d-z1c)+ &
737  (z2s-z1a)/(z1d-z1a)* &
738  (z2s-z1b)/(z1d-z1b)* &
739  1/(z1d-z1c)
740  ENDIF
741  ENDDO
742 ! INTERPOLATE.
743  DO n=1,nm
744  DO i=1,im
745  k1=k1s(i,k2)
746  IF(k1.EQ.0) THEN
747  q2s=q1(1+(i-1)*ixq1+(n-1)*nxq1)
748  j2s=0
749  ELSEIF(k1.EQ.km1) THEN
750  q2s=q1(1+(i-1)*ixq1+(km1-1)*kxq1+(n-1)*nxq1)
751  j2s=0
752  ELSEIF(k1.EQ.1.OR.k1.EQ.km1-1) THEN
753  q1a=q1(1+(i-1)*ixq1+(k1-1)*kxq1+(n-1)*nxq1)
754  q1b=q1(1+(i-1)*ixq1+(k1+0)*kxq1+(n-1)*nxq1)
755  q2s=ffa(i)*q1a+ffb(i)*q1b
756  j2s=gga(i)*q1a+ggb(i)*q1b
757  ELSE
758  q1a=q1(1+(i-1)*ixq1+(k1-2)*kxq1+(n-1)*nxq1)
759  q1b=q1(1+(i-1)*ixq1+(k1-1)*kxq1+(n-1)*nxq1)
760  q1c=q1(1+(i-1)*ixq1+(k1+0)*kxq1+(n-1)*nxq1)
761  q1d=q1(1+(i-1)*ixq1+(k1+1)*kxq1+(n-1)*nxq1)
762  q2s=ffa(i)*q1a+ffb(i)*q1b+ffc(i)*q1c+ffd(i)*q1d
763  j2s=gga(i)*q1a+ggb(i)*q1b+ggc(i)*q1c+ggd(i)*q1d
764  IF(q2s.LT.min(q1b,q1c)) THEN
765  q2s=min(q1b,q1c)
766  j2s=0
767  ELSEIF(q2s.GT.max(q1b,q1c)) THEN
768  q2s=max(q1b,q1c)
769  j2s=0
770  ENDIF
771  ENDIF
772  q2(1+(i-1)*ixq2+(k2-1)*kxq2+(n-1)*nxq2)=q2s
773  j2(1+(i-1)*ixq2+(k2-1)*kxq2+(n-1)*nxq2)=j2s
774  ENDDO
775  ENDDO
776  ENDDO
777 !$OMP END PARALLEL DO
778 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
779  END SUBROUTINE terp3

781  SUBROUTINE rsearch(IM,KM1,IXZ1,KXZ1,Z1,KM2,IXZ2,KXZ2,Z2,IXL2,KXL2,&
782  & l2)
783 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
784 !
785 ! SUBPROGRAM: RSEARCH SEARCH FOR A SURROUNDING REAL INTERVAL
786 ! PRGMMR: IREDELL ORG: W/NMC23 DATE: 98-05-01
787 !
788 ! ABSTRACT: THIS SUBPROGRAM SEARCHES MONOTONIC SEQUENCES OF REAL NUMBERS
789 ! FOR INTERVALS THAT SURROUND A GIVEN SEARCH SET OF REAL NUMBERS.
790 ! THE SEQUENCES MAY BE MONOTONIC IN EITHER DIRECTION; THE REAL NUMBERS
791 ! MAY BE SINGLE OR DOUBLE PRECISION; THE INPUT SEQUENCES AND SETS
792 ! AND THE OUTPUT LOCATIONS MAY BE ARBITRARILY DIMENSIONED.
793 !
794 ! PROGRAM HISTORY LOG:
795 ! 1999-01-05 MARK IREDELL
796 !
797 ! USAGE: CALL RSEARCH(IM,KM1,IXZ1,KXZ1,Z1,KM2,IXZ2,KXZ2,Z2,IXL2,KXL2,
798 ! & L2)
799 ! INPUT ARGUMENT LIST:
800 ! IM INTEGER NUMBER OF SEQUENCES TO SEARCH
801 ! KM1 INTEGER NUMBER OF POINTS IN EACH SEQUENCE
802 ! IXZ1 INTEGER SEQUENCE SKIP NUMBER FOR Z1
803 ! KXZ1 INTEGER POINT SKIP NUMBER FOR Z1
804 ! Z1 REAL (1+(IM-1)*IXZ1+(KM1-1)*KXZ1)
805 ! SEQUENCE VALUES TO SEARCH
806 ! (Z1 MUST BE MONOTONIC IN EITHER DIRECTION)
807 ! KM2 INTEGER NUMBER OF POINTS TO SEARCH FOR
808 ! IN EACH RESPECTIVE SEQUENCE
809 ! IXZ2 INTEGER SEQUENCE SKIP NUMBER FOR Z2
810 ! KXZ2 INTEGER POINT SKIP NUMBER FOR Z2
811 ! Z2 REAL (1+(IM-1)*IXZ2+(KM2-1)*KXZ2)
812 ! SET OF VALUES TO SEARCH FOR
813 ! (Z2 NEED NOT BE MONOTONIC)
814 ! IXL2 INTEGER SEQUENCE SKIP NUMBER FOR L2
815 ! KXL2 INTEGER POINT SKIP NUMBER FOR L2
816 !
817 ! OUTPUT ARGUMENT LIST:
818 ! L2 INTEGER (1+(IM-1)*IXL2+(KM2-1)*KXL2)
819 ! INTERVAL LOCATIONS HAVING VALUES FROM 0 TO KM1
820 ! (Z2 WILL BE BETWEEN Z1(L2) AND Z1(L2+1))
821 !
822 ! SUBPROGRAMS CALLED:
823 ! SBSRCH ESSL BINARY SEARCH
824 ! DBSRCH ESSL BINARY SEARCH
825 !
826 ! REMARKS:
827 ! IF THE ARRAY Z1 IS DIMENSIONED (IM,KM1), THEN THE SKIP NUMBERS ARE
828 ! IXZ1=1 AND KXZ1=IM; IF IT IS DIMENSIONED (KM1,IM), THEN THE SKIP
829 ! NUMBERS ARE IXZ1=KM1 AND KXZ1=1; IF IT IS DIMENSIONED (IM,JM,KM1),
830 ! THEN THE SKIP NUMBERS ARE IXZ1=1 AND KXZ1=IM*JM; ETCETERA.
831 ! SIMILAR EXAMPLES APPLY TO THE SKIP NUMBERS FOR Z2 AND L2.
832 !
833 ! RETURNED VALUES OF 0 OR KM1 INDICATE THAT THE GIVEN SEARCH VALUE
834 ! IS OUTSIDE THE RANGE OF THE SEQUENCE.
835 !
836 ! IF A SEARCH VALUE IS IDENTICAL TO ONE OF THE SEQUENCE VALUES
837 ! THEN THE LOCATION RETURNED POINTS TO THE IDENTICAL VALUE.
838 ! IF THE SEQUENCE IS NOT STRICTLY MONOTONIC AND A SEARCH VALUE IS
839 ! IDENTICAL TO MORE THAN ONE OF THE SEQUENCE VALUES, THEN THE
840 ! LOCATION RETURNED MAY POINT TO ANY OF THE IDENTICAL VALUES.
841 !
842 ! TO BE EXACT, FOR EACH I FROM 1 TO IM AND FOR EACH K FROM 1 TO KM2,
843 ! Z=Z2(1+(I-1)*IXZ2+(K-1)*KXZ2) IS THE SEARCH VALUE AND
844 ! L=L2(1+(I-1)*IXL2+(K-1)*KXL2) IS THE LOCATION RETURNED.
845 ! IF L=0, THEN Z IS LESS THAN THE START POINT Z1(1+(I-1)*IXZ1)
846 ! FOR ASCENDING SEQUENCES (OR GREATER THAN FOR DESCENDING SEQUENCES).
847 ! IF L=KM1, THEN Z IS GREATER THAN OR EQUAL TO THE END POINT
848 ! Z1(1+(I-1)*IXZ1+(KM1-1)*KXZ1) FOR ASCENDING SEQUENCES
849 ! (OR LESS THAN OR EQUAL TO FOR DESCENDING SEQUENCES).
850 ! OTHERWISE Z IS BETWEEN THE VALUES Z1(1+(I-1)*IXZ1+(L-1)*KXZ1) AND
851 ! Z1(1+(I-1)*IXZ1+(L-0)*KXZ1) AND MAY EQUAL THE FORMER.
852 !
853 ! ATTRIBUTES:
854 ! LANGUAGE: FORTRAN
855 !
856 !C$$$
857 ! IMPLICIT NONE
858 ! INTEGER,INTENT(IN):: IM,KM1,IXZ1,KXZ1,KM2,IXZ2,KXZ2,IXL2,KXL2
859 ! REAL,INTENT(IN):: Z1(1+(IM-1)*IXZ1+(KM1-1)*KXZ1)
860 ! REAL,INTENT(IN):: Z2(1+(IM-1)*IXZ2+(KM2-1)*KXZ2)
861 ! INTEGER,INTENT(OUT):: L2(1+(IM-1)*IXL2+(KM2-1)*KXL2)
