39 subroutine eigensym2(em,vv,oo)! [eigensym2]
40 !=============================================================================
41 ! Get the orthogonal eigenvectors, vv, and diagonal matrix of eigenvalues, oo,
42 ! of the symmetric 2*2 matrix, em.
43 !=============================================================================
44 implicit none
45 real(dp),dimension(2,2),intent(in ):: em
46 real(dp),dimension(2,2),intent(out):: vv,oo
47 !-----------------------------------------------------------------------------
48 real(dp):: a,b,c,d,e,f,g,h
49 !=============================================================================
50 a=em(1,1); b=em(1,2); c=em(2,2)
51 d=a*c-b*b! <- det(em)
52 e=(a+c)*o2; f=(a-c)*o2
53 h=sqrt(f**2+b**2)
54 g=sqrt(b**2+(h+abs(f))**2)
55 if (g==u0)then; vv(:,1)=(/u1,u0/)
56 elseif(f> u0)then; vv(:,1)=(/h+f,b/)/g
57 else ; vv(:,1)=(/b,h-f/)/g
58 endif
59 vv(:,2)=(/-vv(2,1),vv(1,1)/)
60 oo=matmul(transpose(vv),matmul(em,vv))
61 oo(1,2)=u0; oo(2,1)=u0
62 end subroutine eigensym2

64 subroutine eigensym2d(em,vv,oo,vvd,ood,ff)! [eigensym2]
65 !=============================================================================
66 ! For a symmetric 2*2 matrix, em, return the normalized eigenvectors, vv, and
67 ! the diagonal matrix of eigenvalues, oo. If the two eigenvalues are equal,
68 ! proceed no further and raise the logical failure flag, ff, to .true.;
69 ! otherwise, return with vvd=d(vv)/d(em) and ood=d(oo)/d(em) and ff=.false.,
70 ! and maintain the symmetries between the last two of the indices of
71 ! these derivatives.
72 !=============================================================================
73 implicit none
74 real(dp),dimension(2,2), intent(in ):: em
75 real(dp),dimension(2,2), intent(out):: vv,oo
76 real(dp),dimension(2,2,2,2),intent(out):: vvd,ood
77 logical, intent(out):: ff
78 !-----------------------------------------------------------------------------
79 real(dp),dimension(2,2):: emd,tt,vvr
80 real(dp) :: oodif,dtheta
81 integer(spi) :: i,j
82 !=============================================================================
83 call eigensym2(em,vv,oo); vvr(1,:)=-vv(2,:); vvr(2,:)=vv(1,:)
84 oodif=oo(1,1)-oo(2,2); ff=oodif==u0; if(ff)return
85 ood=0
86 vvd=0
87 do j=1,2
88  do i=1,2
89  emd=0
90  if(i==j)then
91  emd(i,j)=u1
92  else
93  emd(i,j)=o2; emd(j,i)=o2
94  endif
95  tt=matmul(transpose(vv),matmul(emd,vv))
96  ood(1,1,i,j)=tt(1,1)
97  ood(2,2,i,j)=tt(2,2)
98  dtheta=tt(1,2)/oodif
99  vvd(:,:,i,j)=vvr*dtheta
100  enddo
101 enddo
102 end subroutine eigensym2d

106 !=============================================================================
107 ! Get the inverse of a 2*2 matrix (need not be symmetric in this case).
108 !=============================================================================

120 subroutine invsym2d(em,z,zd)! [invsym2]
121 !=============================================================================
122 ! Get the inverse, z,of a 2*2 symmetric matrix, em, and its derivative, zd,
123 ! with respect to symmetric variations of its components. I.e., for a
124 ! symmetric infinitesimal change, delta_em, in em, the resulting
125 ! infinitesimal change in z would be:
126 ! delta_z(i,j) = matmul(zd(i,j,:,:),delta_em)
127 !=============================================================================
128 implicit none
129 real(dp),dimension(2,2) ,intent(in ):: em
130 real(dp),dimension(2,2) ,intent(out):: z
131 real(dp),dimension(2,2,2,2),intent(out):: zd
132 !-----------------------------------------------------------------------------
133 integer(spi):: k,l
134 !=============================================================================
135 call invsym2(em,z)
136 call id2222(zd)
137 do l=1,2; do k=1,2
138  zd(:,:,k,l)=-matmul(matmul(z,zd(:,:,k,l)),z)
139 enddo; enddo
140 end subroutine invsym2d

162 subroutine sqrtsym2d(x,z,zd)! [sqrtsym2]
163 !=============================================================================
164 ! General routine to evaluate z=sqrt(x), and the symmetric
165 ! derivative, zd = dz/dx, where x is a symmetric 2*2 positive-definite
166 ! matrix. If the eigenvalues are very close together, extract their
167 ! geometric mean for "preconditioning" a scaled version, px, of x, whose
168 ! sqrt, and hence its derivative, can be easily obtained by the series
169 ! expansion method. Otherwise, use the eigen-method (which entails dividing
170 ! by the difference in the eignevalues to get zd, and which therefore
171 ! fails when the eigenvalues become too similar).
172 !=============================================================================
173 implicit none
174 real(dp),dimension(2,2), intent(in ):: x
175 real(dp),dimension(2,2), intent(out):: z
176 real(dp),dimension(2,2,2,2),intent(out):: zd
177 !-----------------------------------------------------------------------------
178 real(dp),dimension(2,2):: px
179 real(dp) :: rdetx,lrdetx,htrpxs,q
180 !=============================================================================
181 rdetx=sqrt(x(1,1)*x(2,2)-x(1,2)*x(2,1)) ! <- sqrt(determinant of x)
182 lrdetx=sqrt(rdetx)
183 px=x/rdetx ! <- preconditioned x (has unit determinant)
184 htrpxs= ((px(1,1)+px(2,2))/2)**2 ! <- {half-trace-px}-squared
185 q=htrpxs-u1
186 if(q<.05_dp)then ! <- Taylor expansion method
187  call sqrtsym2d_t(px,z,zd)
188  z=z*lrdetx; zd=zd/lrdetx
189 else
190  call sqrtsym2d_e(x,z,zd) ! <- Eigen-method
191 endif
192 end subroutine sqrtsym2d

220 !=============================================================================
221 subroutine sqrtsym2d_t(x,z,zd)! [sqrtsym2d_t]
222 !=============================================================================
223 ! Use the Taylor-series method (eigenvalues both fairly close to unity).
224 ! For a 2*2 positive definite symmetric matrix x, try to get both the z=sqrt(x)
225 ! and dz/dx using the binomial-expansion method applied to the intermediate
226 ! matrix, r = (x-1). ie z=sqrt(x) = (1+r)^{1/2} = I + (1/2)*r -(1/8)*r^2 ...
227 ! + [(-)^n *(2n)!/{(n+1)! * n! *2^{2*n-1}} ]*r^{n+1}
228 !=============================================================================
229 implicit none
230 real(dp),dimension(2,2), intent(in ):: x
231 real(dp),dimension(2,2), intent(out):: z
232 real(dp),dimension(2,2,2,2),intent(out):: zd
233 !-----------------------------------------------------------------------------
234 integer(spi),parameter :: nit=300 ! number of iterative increments allowed
235 real(dp),parameter :: crit=1.e-17
236 real(dp),dimension(2,2) :: r,rp,rd,rpd
237 real(dp) :: c
238 integer(spi) :: i,j,n
239 !=============================================================================
240 r=x; r(1,1)=x(1,1)-1; r(2,2)=x(2,2)-1
241 z=u0; z(1,1)=u1; z(2,2)=u1
242 rp=r
243 c=o2
244 do n=0,nit
245  z=z+c*rp
246  rp=matmul(rp,r)
247  if(sum(abs(rp))<crit)exit
248  c=-c*(n*2+1)/(2*(n+2))
249 enddo
250 do j=1,2; do i=1,2
251  rd=id(:,:,i,j)
252  rpd=rd
253  zd(:,:,i,j)=0
254  rp=r
255  c=o2
256  do n=0,nit
257  zd(:,:,i,j)=zd(:,:,i,j)+c*rpd
258  rpd=matmul(rd,rp)+matmul(r,rpd)
259  rp=matmul(r,rp)
260  if(sum(abs(rp))<crit)exit
261  c=-c*(n*2+1)/(2*(n+2))
262  enddo
263 enddo; enddo
264 end subroutine sqrtsym2d_t

329 subroutine expsym2d_t(x,z,zd)! [expsym2d_t]
330 !=============================================================================
331 ! Use the Taylor-series method (eigenvalues both fairly close to zero).
332 ! For a 2*2 symmetric matrix x, try to get both the z=exp(x)
333 ! and dz/dx using the Taylor series expansion method.
334 !=============================================================================
335 implicit none
336 real(dp),dimension(2,2), intent(in ):: x
337 real(dp),dimension(2,2), intent(out):: z
338 real(dp),dimension(2,2,2,2),intent(out):: zd
339 !-----------------------------------------------------------------------------
340 integer(spi),parameter :: nit=100 ! number of iterative increments allowed
341 real(dp),parameter :: crit=1.e-17_dp
342 real(dp),dimension(2,2) :: xp,xd,xpd
343 real(dp) :: c
344 integer(spi) :: i,j,n
345 !=============================================================================
346 z=0; z(1,1)=u1; z(2,2)=u1
347 xp=x
348 c=u1
349 do n=1,nit
350  z=z+c*xp
351  xp=matmul(xp,x)
352  if(sum(abs(xp))<crit)exit
353  c=c/(n+1)
354 enddo
355 do j=1,2; do i=1,2
356  xd=id(:,:,i,j)
357  xpd=xd
358  zd(:,:,i,j)=0
359  xp=x
360  c=u1
361  do n=1,nit
362  zd(:,:,i,j)=zd(:,:,i,j)+c*xpd
363  xpd=matmul(xd,xp)+matmul(x,xpd)
364  xp=matmul(x,xp)
365  if(sum(abs(xpd))<crit)exit
366  c=c/(n+1)
367  enddo
368 enddo; enddo
369 end subroutine expsym2d_t

391 subroutine logsym2d(x,z,zd)! [logsym2]
392 !=============================================================================
393 ! General routine to evaluate the logarithm, z=log(x), and the symmetric
394 ! derivative, zd = dz/dx, where x is a symmetric 2*2 positive-definite
395 ! matrix.
396 !=============================================================================
397 use pfun, only: sinhox
398 implicit none
399 real(dp),dimension(2,2), intent(in ):: x
400 real(dp),dimension(2,2), intent(out):: z
401 real(dp),dimension(2,2,2,2),intent(out):: zd
402 !-----------------------------------------------------------------------------
403 real(dp),dimension(2,2):: vv,oo,d11,d12,d22,pqr
404 real(dp) :: c,s,cc,cs,ss,c2h,p,q,r,lp,lq,l
405 integer(spi) :: i
406 !=============================================================================
407 call eigensym2(x,vv,oo)
408 if(oo(1,1)<=u0 .or. oo(2,2)<=u0)stop 'In logsym2; matrix x is not positive definite'
409 c=vv(1,1); s=vv(1,2); cc=c*c; cs=c*s; ss=s*s; c2h=(cc-ss)*o2
410 p=u1/oo(1,1); q=u1/oo(2,2); lp=log(p); lq=log(q); oo(1,1)=-lp; oo(2,2)=-lq
411 z=matmul(vv,matmul(oo,transpose(vv)))
412 l=(lp-lq)*o2; r=sqrt(p*q)/sinhox(l)
413 d11(1,:)=(/ cc,cs /); d11(2,:)=(/ cs,ss/)
414 d12(1,:)=(/-cs,c2h/); d12(2,:)=(/c2h,cs/)
415 d22(1,:)=(/ ss,-cs/); d22(2,:)=(/-cs,cc/)
416 pqr(1,:)=(/p,r/) ; pqr(2,:)=(/r,q/)
417 zd(:,:,1,1)=matmul(vv,matmul(d11*pqr,transpose(vv)))
418 zd(:,:,1,2)=matmul(vv,matmul(d12*pqr,transpose(vv)))
419 zd(:,:,2,2)=matmul(vv,matmul(d22*pqr,transpose(vv)))
420 zd(:,:,2,1)=zd(:,:,1,2)
421 end subroutine logsym2d

434 subroutine chol2(s,c,ff)! [chol2]
435 !===========================================================================
436 ! Return the cholesky lower triangular factor, C, of the 2X2 symmetric
437 ! matrix, S, or raise the failure flag, FF, if S is not positive-definite.
438 !===========================================================================
439 use pietc, only: u0
440 implicit none
441 real(dp),dimension(2,2),intent(in ):: s
442 real(dp),dimension(2,2),intent(out):: c
443 logical ,intent(out):: ff
444 !---------------------------------------------------------------------------
445 real(dp):: r
446 !===========================================================================
447 ff=s(1,1)<=u0; if(ff)return
448 c(1,1)=sqrt(s(1,1))
449 c(1,2)=u0
450 c(2,1)=s(2,1)/c(1,1)
451 r=s(2,2)-c(2,1)**2
452 ff=r<=u0; if(ff)return
453 c(2,2)=sqrt(r)
454 end subroutine chol2
