97 subroutine sinvmtf(a,ff)! [inv]
98 !=============================================================================
99 ! Invert matrix (or flag if can't)
100 !----------------
101 use pietc_s, only: u1
102 real(sp),dimension(:,:),intent(inout):: a
103 logical, intent( out):: ff
104 integer(spi) :: m,i,j,jp,l
105 real(sp) :: d
106 integer(spi),dimension(size(a,1)):: ipiv
107 !=============================================================================
108 m=size(a,1)
109 if(m /= size(a,2))stop 'In sinvmtf; matrix passed to sinvmtf is not square'
110 ! Perform a pivoted L-D-U decomposition on matrix a:
111 call sldumf(a,ipiv,d,ff)
112 if(ff)then
113  print '(" In sinvmtf; failed call to sldumf")'
114  return
115 endif
116 
117 ! Invert upper triangular portion U in place:
118 do i=1,m; a(i,i)=u1/a(i,i); enddo
119 do i=1,m-1
120  do j=i+1,m; a(i,j)=-a(j,j)*dot_product(a(i:j-1,j),a(i,i:j-1)); enddo
121 enddo
122 
123 ! Invert lower triangular portion L in place:
124 do j=1,m-1; jp=j+1
125  do i=jp,m; a(i,j)=-a(i,j)-dot_product(a(jp:i-1,j),a(i,jp:i-1)); enddo
126 enddo
127 
128 ! Form the product of U**-1 and L**-1 in place
129 do j=1,m-1; jp=j+1
130  do i=1,j; a(i,j)=a(i,j)+dot_product(a(jp:m,j),a(i,jp:m)); enddo
131  do i=jp,m; a(i,j)=dot_product(a(i:m,j),a(i,i:m)); enddo
132 enddo
133 
134 ! Permute columns according to ipiv
135 do j=m-1,1,-1; l=ipiv(j); call sswpvv(a(:,j),a(:,l)); enddo
136 end subroutine sinvmtf

438 subroutine sldumf(a,ipiv,d,ff)! [ldum]
439 !=============================================================================
440 ! R.J.Purser, NCEP, Washington D.C. 1996
441 ! SUBROUTINE LDUM
442 ! perform l-d-u decomposition of square matrix a in place with
443 ! pivoting.
444 !
445 ! <-> a square matrix to be factorized
446 ! <-- ipiv array encoding the pivoting sequence
447 ! <-- d indicator for possible sign change of determinant
448 ! <-- ff: failure flag, set to .true. when determinant of a vanishes.
449 !=============================================================================
450 use pietc_s,only: u0,u1
451 real(sp), intent(inout) :: a(:,:)
452 real(sp), intent( out) :: d
453 integer(spi),intent( out) :: ipiv(:)
454 logical, intent( out) :: ff
455 integer(spi):: m,i, j, jp, ibig, jm
456 real(sp) :: s(size(a,1)), aam, aa, abig, ajj, ajji, aij
457 !=============================================================================
458 ff=f
459 m=size(a,1)
460 do i=1,m
461  aam=u0
462  do j=1,m
463  aa=abs(a(i,j))
464  if(aa > aam)aam=aa
465  enddo
466  if(aam == u0)then
467  print '("In sldumf; row ",i6," of matrix vanishes")',i
468  ff=t
469  return
470  endif
471  s(i)=u1/aam
472 enddo
473 d=1_sp
474 ipiv(m)=m
475 do j=1,m-1
476  jp=j+1
477  abig=s(j)*abs(a(j,j))
478  ibig=j
479  do i=jp,m
480  aa=s(i)*abs(a(i,j))
481  if(aa > abig)then
482  ibig=i
483  abig=aa
484  endif
485  enddo
486 ! swap rows, recording changed sign of determinant
487  ipiv(j)=ibig
488  if(ibig /= j)then
489  d=-d
490  call sswpvv(a(j,:),a(ibig,:))
491  s(ibig)=s(j)
492  endif
493  ajj=a(j,j)
494  if(ajj == u0)then
495  jm=j-1
496  print '(" failure in sldumf:"/" matrix singular, rank=",i3)',jm
497  ff=t
498  return
499  endif
500  ajji=u1/ajj
501  do i=jp,m
502  aij=ajji*a(i,j)
503  a(i,j)=aij
504  a(i,jp:m) = a(i,jp:m) - aij*a(j,jp:m)
505  enddo
506 enddo
507 end subroutine sldumf
