512 function sarea_s(v1,v2,v3)result(area)! [sarea]
513 !=============================================================================
514 ! Compute the area of the spherical triangle, {v1,v2,v3}, measured in the
515 ! right-handed sense, by dropping a perpendicular to u0 on the longest side so
516 ! that the area becomes the sum of areas of the two simpler right-angled
517 ! triangles.
518 !=============================================================================
519 use pietc_s, only: zero=>u0
520 implicit none
521 real(sp),dimension(3),intent(IN ):: v1,v2,v3
522 real(sp) :: area
523 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
524 real(sp) :: s123,a1,a2,b,d1,d2,d3
525 real(sp),dimension(3) :: u0,u1,u2,u3,x,y
526 !=============================================================================
527 area=zero
528 u1=normalized(v1); u2=normalized(v2); u3=normalized(v3)
529 s123=triple_product(u1,u2,u3)
530 if(s123==zero)return
531 
532 d1=dot_product(u3-u2,u3-u2)
533 d2=dot_product(u1-u3,u1-u3)
534 d3=dot_product(u2-u1,u2-u1)
535 
536 ! Triangle that is not degenerate. Cyclically permute, so side 3 is longest:
537 if(d3<d1 .or. d3<d2)call cyclic(u1,u2,u3,d1,d2,d3)
538 if(d3<d1 .or. d3<d2)call cyclic(u1,u2,u3,d1,d2,d3)
539 y=normalized( cross_product(u1,u2) )
540 b=dot_product(y,u3)
541 u0=normalized( u3-y*b )
542 x=cross_product(y,u0)
543 a1=-dot_product(x,u1-u0); a2= dot_product(x,u2-u0)
544 area=huarea(a1,b)+huarea(a2,b)
545

549 !-----------------------------------------------------------------------------

552  real(sp), intent(INOUT):: d1,d2,d3

555  dt=d1; d1=d2; d2=d3; d3=dt
556  ut=u1; u1=u2; u2=u3; u3=ut
557  end subroutine cyclic

604 !=============================================================================

606 ! rlat, and two other vertices, A and B, whose incremented latitudes
607 ! and longitudes are drlata,drlona (for A) and drlatb,drlonb (for B).
608 ! The computations are designed to give a proportionately accurate area
609 ! estimate even when the triangle is very small, provided the B-increment
610 ! is not disproportionately small compared to the other two sides.
611 !=============================================================================

1041 !=============================================================================
1042 subroutine rowgram(m,n,a,ipiv,tt,b,rank)! [gram]
1043 !=============================================================================
1044 ! Without changing (tall) rectangular input matrix a, perform pivoted gram-
1045 ! Schmidt operations to orthogonalize the rows, until rows that remain become
1046 ! negligible. Record the pivoting sequence in ipiv, and the row-normalization
1047 ! in tt(j,j) and the row-orthogonalization in tt(i,j), for i>j. Note that
1048 ! tt(i,j)=0 for i<j (tt is truncated lower triangular). The orthonormalized
1049 ! rows are returned in square array b, which is complete even when the
1050 ! effective rank < n.
1051 ! The recorded row operations can be repeated on independent column vectors
1052 ! through the use of subroutine ROWOPS (in this module).
1053 ! It is recommended to rescale the original matrix A via a call to CORRAL
1054 ! (in this module) because the negligibility criterion depends upon an
1055 ! "epsilon" value that is fixed (10**(-13)) and assumes elements of a are
1056 ! never too different in magnitude from unity, unless they are actually zero.
1057 !=============================================================================
1058 use pietc, only: u0,u1
1059 implicit none
1060 integer(spi), intent(IN ):: m,n
1061 real(dp),dimension(m,n), intent(in ):: a
1062 integer(spi),dimension(n),intent(out):: ipiv
1063 real(dp),dimension(m,n), intent(out):: tt
1064 real(dp),dimension(n,n), intent(out):: b
1065 integer(spi), intent(out):: rank
1066 !-----------------------------------------------------------------------------
1067 real(dp),parameter :: eps=1.e-13_dp,epss=eps**2
1068 real(dp),dimension(m,n) :: aa
1069 real(dp),dimension(n) :: rowv
1070 real(dp),dimension(m) :: p
1071 real(dp) :: maxp,nepss
1072 integer(spi),dimension(1):: jloc
1073 integer(spi) :: i,ii,iii,j,maxi
1074 !=============================================================================
1075 if(m<n)stop 'In rowgram; this routines needs m>=n please'
1076 nepss=n*epss
1077 rank=n
1078 aa=a
1079 tt=u0
1080 do ii=1,n
1081 
1082 ! At this stage, all rows less than ii are already orthonormalized and are
1083 ! orthogonal to all rows at and beyond ii. Find the norms of these lower
1084 ! rows and pivot the largest of them into position ii:
1085  maxp=u0
1086  maxi=ii
1087  do i=ii,m
1088  p(i)=dot_product(aa(i,:),aa(i,:))
1089  if(p(i)>maxp)then
1090  maxp=p(i)
1091  maxi=i
1092  endif
1093  enddo
1094  if(maxp<nepss)then
1095  b=u0
1096  b(1:ii-1,:)=aa(1:ii-1,:)
1097 ! fill the remaining rows, ii:n, of b with remaining orthonormal rows at random
1098  do iii=ii,n
1099 ! find the column of b for which the maximum element is the smallest:
1100  do j=1,n
1101  rowv(j)=maxval(abs(b(1:iii-1,j)))
1102  enddo
1103  jloc=minloc(rowv)
1104  j=jloc(1)
1105  b(iii,j)=u1
1106  do i=1,iii-1
1107  maxp=dot_product(b(i,:),b(iii,:))
1108  b(iii,:)=b(iii,:)-b(i,:)*maxp
1109  enddo
1110  maxp=sqrt(dot_product(b(iii,:),b(iii,:)))
1111  b(iii,:)=b(iii,:)/maxp
1112  enddo
1113  rank=ii-1
1114  return
1115  endif
1116 
1117  ipiv(ii)=maxi
1118  if(maxi/=ii)then
1119  rowv =aa(ii, :)
1120  aa(ii, :)=aa(maxi,:)
1121  aa(maxi,:)=rowv
1122  endif
1123  maxp=sqrt(maxp)
1124  tt(ii,ii)=maxp

1168 subroutine corral(m,n,mask,a,d,aa,e)! [corral]
1169 !=============================================================================
1170 ! Find positive diagonals D and E and a Lagrange multiplier F that minimize
1171 ! the row-sum +column-sum of masked terms,
1172 ! (D_i +log(|A_ij|) +E_j)^2
1173 ! subject to the single constraint, sum_j E_j =0, where the mask permits
1174 ! only nonnegligible A_ij to participate in the quadratic quantities.
1175 ! Once a solution for D and E is found, return their exponentials, d and e,
1176 ! together with the rescaled matrix aa such that a = d.aa.e when d and e are
1177 ! interpreted as diagonal matrices.
1178 !=============================================================================
1179 use pietc, only: u0,u1
1180 use pmat, only: inv
1181 implicit none
1182 integer(spi), intent(in ):: m,n
1183 logical, dimension(m,n),intent(in ):: mask
1184 real(dp),dimension(m,n),intent(in ):: a
1185 real(dp),dimension(m ),intent(out):: d
1186 real(dp),dimension(m,n),intent(out):: aa
1187 real(dp),dimension( n),intent(out):: e
1188 !-----------------------------------------------------------------------------
1189 real(dp),dimension(0:m+n,0:m+n):: g
1190 real(dp),dimension(0:m+n) :: h
1191 integer(spi) :: i,j,k,nh
1192 !=============================================================================
1193 nh=1+m+n
1194 aa=u0
1195 do j=1,n
1196 do i=1,m
1197  if(mask(i,j))aa(i,j)=log(abs(a(i,j)))
1198 enddo
1199 enddo
1200 
1201 h=u0
1202 g=u0
1203 
1204 ! Equations on row 0 enforcing Lagrange multiplier F-constraint:
1205 do j=1,n
1206  k=m+j
1207  g(0,k)=u1
1208 enddo
1209 
1210 ! Equations on rows 1:m minimizing row sums of quadratic terms:
1211 do i=1,m
1212  do j=1,n
1213  k=m+j
1214  if(mask(i,j))then
1215  g(i,i)=g(i,i)-u1
1216  g(i,k)=-u1
1217  h(i)=h(i)-aa(i,j)
1218  endif
1219  enddo
1220 enddo
1221 
1222 ! Equations on rows m+1:m+n minimizing col sums subject to constraint
1223 do j=1,n
1224  k=m+j
1225  g(k,0)=u1
1226  do i=1,m
1227  if(mask(i,j))then
1228  g(k,k)=g(k,k)-u1
1229  g(k,i)=-u1
1230  h(k)=h(k)-aa(i,j)
1231  endif
1232  enddo
1233 enddo
1234 
1235 ! Invert the normal equations:
1236 call inv(g,h)
1237 
1238 ! Exponentiate the parts that become final scaling diagnonal matrices d and e:
1239 do i=1,m
1240  d(i)=exp(h(i))
1241 enddo
1242 do j=1,n
1243  k=m+j
1244  e(j)=exp(h(k))
1245 enddo
1246 
1247 ! Compute the rescaled matrix directly:
1248 do j=1,n
1249 do i=1,m
1250  aa(i,j)=a(i,j)/(d(i)*e(j))
1251 enddo
1252 enddo
1253 end subroutine corral

1256 subroutine rottoax(orth33,ax3)! [rottoax]
1257 !=============================================================================
1258 ! Assuming that given orth33 is a 3*3 proper rotation matrix, derive an axial
1259 ! 3-vector, ax3, such that orth33 is implied by ax3 when the latter is
1260 ! interpreted as encoding a rotation (as in subroutine axtorot). Note that
1261 ! such ax3 are not unique -- adding any multiple of 2*pi times the parallel
1262 ! unit vector leads to the same orth33.
1263 !=============================================================================
1264 implicit none
1265 real(dp),dimension(3,3),intent(in ):: orth33
1266 real(dp),dimension(3), intent(out):: ax3
1267 !-----------------------------------------------------------------------------
1268 real(dp),dimension(3,3) :: plane
1269 real(dp),dimension(3) :: x,y,z
1270 real(dp) :: s
1271 integer(spi),dimension(1):: ii
1272 integer(spi) :: i,j,k
1273 !=============================================================================
1274 plane=orth33-identity()! Columns must be coplanar vectors
1275 do i=1,3; z(i)=dot_product(plane(:,i),plane(:,i)); enddo
1276 ii=minloc(z)
1277 k=ii(1); i=1+mod(k,3); j=1+mod(i,3)
1278 ax3=cross_product(plane(:,i),plane(:,j))
1279 s=absv(ax3); if(s==0)return
1280 ax3=ax3/s ! <- temporarily a unit vector pointing along rotation axis
1281 ! Construct a unit 2D basis, x,y, in the plane of rotation
1282 x=normalized(cross_product(ax3,plane(:,j)))
1283 y=cross_product(ax3,x)
1284 z=matmul(orth33,x)! Rotate x by the original rotation matrix
1285 ax3=ax3*atan2(dot_product(y,z),dot_product(x,z))! multiply ax3 by the angle
1286 end subroutine rottoax

1334 subroutine rottoq(rot,q)! [rottoq]
1335 !=============================================================================
1336 ! Go from rotation matrix to quaternion representation
1337 !=============================================================================
1338 use pietc, only: zero=>u0,one=>u1,two=>u2
1339 implicit none
1340 real(dp),dimension(3,3),intent(IN ):: rot
1341 real(dp),dimension(0:3),intent(OUT):: q
1342 !------------------------------------------------------------------------------
1343 real(dp),dimension(3,3) :: t1,t2
1344 real(dp),dimension(3) :: u1,u2
1345 real(dp) :: gamma,gammah,s,ss
1346 integer(spi) :: i,j
1347 integer(spi),dimension(1):: ii
1348 !==============================================================================
1349 ! construct the orthogonal matrix, t1, whose third row is the rotation axis
1350 ! of rot:
1351 t1=rot; do i=1,3; t1(i,i)=t1(i,i)-1; u1(i)=dot_product(t1(i,:),t1(i,:)); enddo
1352 ii=maxloc(u1); j=ii(1); ss=u1(j)
1353 if(ss<1.e-16_dp)then
1354  q=zero; q(0)=one; return
1355 endif
1356 t1(j,:)=t1(j,:)/sqrt(ss)
1357 if(j/=1)then
1358  u2 =t1(1,:)
1359  t1(1,:)=t1(j,:)
1360  t1(j,:)=u2
1361 endif
1362 do i=2,3
1363  t1(i,:)=t1(i,:)-dot_product(t1(1,:),t1(i,:))*t1(1,:)
1364  u1(i)=dot_product(t1(i,:),t1(i,:))
1365 enddo
1366 if(u1(3)>u1(2))then
1367  j=3
1368 else
1369  j=2
1370 endif
1371 ss=u1(j)
1372 if(ss==zero)stop 'In rotov; invalid rot'
1373 if(j/=2)t1(2,:)=t1(3,:)
1374 t1(2,:)=t1(2,:)/sqrt(ss)
1375 
1376 ! Form t1(3,:) as the cross product of t1(1,:) and t1(2,:)
1377 t1(3,1)=t1(1,2)*t1(2,3)-t1(1,3)*t1(2,2)
1378 t1(3,2)=t1(1,3)*t1(2,1)-t1(1,1)*t1(2,3)
1379 t1(3,3)=t1(1,1)*t1(2,2)-t1(1,2)*t1(2,1)
1380 
1381 ! Project rot into the frame whose axes are the rows of t1:
1382 t2=matmul(t1,matmul(rot,transpose(t1)))
1383 
1384 ! Obtain the rotation angle, gamma, implied by rot, and gammah=gamma/2:
1385 gamma=atan2(t2(2,1),t2(1,1)); gammah=gamma/two
1386 
1387 ! Hence deduce coefficients (in the form of a real 4-vector) of one of the two
1388 ! possible equivalent spinors:
1389 s=sin(gammah)
1390 q(0)=cos(gammah)
1391 q(1:3)=t1(3,:)*s
1392 end subroutine rottoq
