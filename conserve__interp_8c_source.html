413  mpp_gather_field_double(interp[n].nxgrid, interp[n].area, gdata_dbl);
414  mpp_put_var_value(fid, id_xgrid_area, gdata_dbl);
415 
416  if(opcode & CONSERVE_ORDER2) {
417  start[1] = 0;
418  mpp_gather_field_double(interp[n].nxgrid, interp[n].di_in, gdata_dbl);
419  mpp_put_var_value_block(fid, id_tile1_dist, start, nwrite, gdata_dbl);
420  start[1] = 1;
421  mpp_gather_field_double(interp[n].nxgrid, interp[n].dj_in, gdata_dbl);
422  mpp_put_var_value_block(fid, id_tile1_dist, start, nwrite, gdata_dbl);
423  }
424 
425  free(gdata_dbl);
426  mpp_close(fid);
427  }
428  }
429  }
430  if(mpp_pe() == mpp_root_pe())printf("NOTE: done calculating index and weight for conservative interpolation\n");
431  }
432 
433  /* check the input area match exchange grid area */
434  if(opcode & CHECK_CONSERVE) {
435  int nx1, ny1, max_i, max_j, i, j;
436  double max_ratio, ratio_change;
437  double *area2;
438 
439  /* sum over exchange grid to get the area of grid_in */
440  nx1 = grid_out[0].nxc;
441  ny1 = grid_out[0].nyc;
442 
443  area2 = (double *)malloc(nx1*ny1*sizeof(double));
444 
445  for(n=0; n<ntiles_out; n++) {
446  for(i=0; i<nx1*ny1; i++) area2[i] = 0;
447  for(i=0; i<interp[n].nxgrid; i++) {
448  ii = interp[n].j_out[i]*nx1 + interp[n].i_out[i];
449  area2[ii] += interp[n].area[i];
450  }
451  max_ratio = 0;
452  max_i = 0;
453  max_j = 0;
454  /* comparing area1 and area2 */
455  for(j=0; j<ny1; j++) for(i=0; i<nx1; i++) {
456  ii = j*nx1+i;
457  ratio_change = fabs(grid_out[n].cell_area[ii]-area2[ii])/grid_out[n].cell_area[ii];
458  if(ratio_change > max_ratio) {
459  max_ratio = ratio_change;
460  max_i = i;
461  max_j = j;
462  }
463  if( ratio_change > 1.e-4 ) {
464  printf("(i,j)=(%d,%d), change = %g, area1=%g, area2=%g\n", i, j, ratio_change, grid_out[n].cell_area[ii],area2[ii]);
465  }
466  }
467  ii = max_j*nx1+max_i;
468  printf("The maximum ratio change at (%d,%d) = %g, area1=%g, area2=%g\n", max_i, max_j, max_ratio, grid_out[n].cell_area[ii],area2[ii]);
469 
470  }
471 
472  free(area2);
473 
474  }
475 
476  free(i_in);
477  free(j_in);
478  free(i_out);
479  free(j_out);
480  free(xgrid_area);
481  if(xgrid_clon) free(xgrid_clon);
482  if(xgrid_clat) free(xgrid_clat);
483

487 /*******************************************************************************
488  void do_scalar_conserve_interp( )
489  doing conservative interpolation
490 *******************************************************************************/
491 void do_scalar_conserve_interp(Interp_config *interp, int varid, int ntiles_in, const Grid_config *grid_in,
492  int ntiles_out, const Grid_config *grid_out, const Field_config *field_in,
493  Field_config *field_out, unsigned int opcode, int nz)
494 {

496  int k, n0;
497  int has_missing, halo, interp_method;
498  int weight_exist;
499  int cell_measures, cell_methods;
500  double area, missing, di, dj, area_missing;
501  double *out_area;
502  double gsum_out;
503  int monotonic;
504  Monotone_config *monotone_data;
505 
506  gsum_out = 0;
507  interp_method = field_in->var[varid].interp_method;
508  halo = 0;
509  monotonic = 0;
510  if(interp_method == CONSERVE_ORDER2) {
511  halo = 1;
512  monotonic = opcode & MONOTONIC;
513  }
514 
515  area_missing = field_in->var[varid].area_missing;
516  has_missing = field_in->var[varid].has_missing;
517  weight_exist = grid_in[0].weight_exist;
518  cell_measures = field_in->var[varid].cell_measures;
519  cell_methods = field_in->var[varid].cell_methods;
520 
521  missing = -MAXVAL;
522  if(has_missing) missing = field_in->var[varid].missing;
523 
524  if( nz>1 && has_missing ) mpp_error("conserve_interp: has_missing should be false when nz > 1");
525  if( nz>1 && cell_measures ) mpp_error("conserve_interp: cell_measures should be false when nz > 1");
526  if( nz>1 && cell_methods == CELL_METHODS_SUM ) mpp_error("conserve_interp: cell_methods should not be sum when nz > 1");
527  /* if( nz>1 && monotonic ) mpp_error("conserve_interp: monotonic should be false when nz > 1"); */
528 
529  if(monotonic) monotone_data = (Monotone_config *)malloc(ntiles_in*sizeof(Monotone_config));
530 
531  for(m=0; m<ntiles_out; m++) {
532  nx2 = grid_out[m].nxc;
533  ny2 = grid_out[m].nyc;
534  out_area = (double *)malloc(nx2*ny2*nz*sizeof(double));
535 
536  for(i=0; i<nx2*ny2*nz; i++) field_out[m].data[i] = 0.0;
537 
538  for(i=0; i<nx2*ny2; i++) out_area[i] = 0.0;
539  if(interp_method == CONSERVE_ORDER1) {
540  if(has_missing) {
541  for(n=0; n<interp[m].nxgrid; n++) {
542  i2 = interp[m].i_out[n];
543  j2 = interp[m].j_out[n];
544  i1 = interp[m].i_in [n];
545  j1 = interp[m].j_in [n];
546  tile = interp[m].t_in [n];
547  area = interp[m].area [n];
548  nx1 = grid_in[tile].nx;
549  ny1 = grid_in[tile].ny;
550  if(weight_exist) area *= grid_in[tile].weight[j1*nx1+i1];
551  n1 = j1*nx1+i1;
552  n0 = j2*nx2+i2;
553 
554  if( field_in[tile].data[n1] != missing ) {
555  if( cell_measures )
556  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
557  else if( cell_methods == CELL_METHODS_SUM )
558  area /= grid_in[tile].cell_area[n1];
559  field_out[m].data[n0] += (field_in[tile].data[n1]*area);
560  out_area[n0] += area;
561  }
562  else if(cell_measures) {
563  if( field_in[tile].area[n1] != area_missing ){
564  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
565  out_area[n0] += area;
566  }
567  }
568  }
569  }
570  else {
571  for(n=0; n<interp[m].nxgrid; n++) {
572  i2 = interp[m].i_out[n];
573  j2 = interp[m].j_out[n];
574  i1 = interp[m].i_in [n];
575  j1 = interp[m].j_in [n];
576  tile = interp[m].t_in [n];
577  area = interp[m].area [n];
578  nx1 = grid_in[tile].nx;
579  ny1 = grid_in[tile].ny;
580  if(weight_exist) area *= grid_in[tile].weight[j1*nx1+i1];
581  for(k=0; k<nz; k++) {
582  n1 = k*nx1*ny1 + j1*nx1+i1;
583  n0 = k*nx2*ny2 + j2*nx2+i2;
584  if( cell_measures )
585  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
586  else if( cell_methods == CELL_METHODS_SUM )
587  area /= grid_in[tile].cell_area[n1];
588  field_out[m].data[n0] += (field_in[tile].data[n1]*area);
589  out_area[n0] += area;
590  }
591  }
592  }
593  }
594  else if(monotonic) {
595  int ii, jj;
596  double f_bar;
597  double *xdata;
598  for(n=0; n<ntiles_in; n++) {
599  nx1 = grid_in[n].nx;
600  ny1 = grid_in[n].ny;
601  monotone_data[n].f_bar_max = (double *)malloc(nx1*ny1*sizeof(double));
602  monotone_data[n].f_bar_min = (double *)malloc(nx1*ny1*sizeof(double));
603  monotone_data[n].f_max = (double *)malloc(nx1*ny1*sizeof(double));
604  monotone_data[n].f_min = (double *)malloc(nx1*ny1*sizeof(double));
605  for(j=0; j<ny1; j++) for(i=0; i<nx1; i++) {
606  n1 = j*nx1+i;
607 
608  monotone_data[n].f_bar_max[n1] = -MAXVAL;
609  monotone_data[n].f_bar_min[n1] = MAXVAL;
610  monotone_data[n].f_max[n1] = -MAXVAL;
611  monotone_data[n].f_min[n1] = MAXVAL;
612  n1 = j*nx1+i;
613  for(jj=j-1; jj<=j+1; jj++) for(ii=i-1; ii<=i+1; ii++) {
614  n2 = (jj+1)*(nx1+2)+ii+1;
615  if( field_in[n].data[n2] != missing ) {
616  if( field_in[n].data[n2] > monotone_data[n].f_bar_max[n1] ) monotone_data[n].f_bar_max[n1] = field_in[n].data[n2];
617  if( field_in[n].data[n2] < monotone_data[n].f_bar_min[n1] ) monotone_data[n].f_bar_min[n1] = field_in[n].data[n2];
618  }
619  }
620  }
621  }
622 
623  xdata = (double *)malloc(interp[m].nxgrid*sizeof(double));
624  for(n=0; n<interp[m].nxgrid; n++) {
625  i1 = interp[m].i_in [n];
626  j1 = interp[m].j_in [n];
627  di = interp[m].di_in[n];
628  dj = interp[m].dj_in[n];
629  tile = interp[m].t_in [n];
630  n1 = j1*nx1+i1;
631  n2 = (j1+1)*(nx1+2)+i1+1;
632  if( field_in[tile].data[n2] != missing ) {
633  if( field_in[tile].grad_mask[n1] ) { /* use zero gradient */
634  xdata[n] = field_in[tile].data[n2];
635  }
636  else {
637  xdata[n] = field_in[tile].data[n2]+field_in[tile].grad_x[n1]*di+field_in[tile].grad_y[n1]*dj;
638  }
639  if(monotonic) {
640  if( xdata[n] > monotone_data[tile].f_max[n1]) monotone_data[tile].f_max[n1] = xdata[n];
641  if( xdata[n] < monotone_data[tile].f_min[n1]) monotone_data[tile].f_min[n1] = xdata[n];
642  }
643  }
644  else
645  xdata[n] = missing;
646  }
647 
648  /* get the global f_max and f_min */
649  if(mpp_npes() >1) {
650  for(n=0; n<ntiles_in; n++) {
651  mpp_min_double(nx1*ny1, monotone_data[n].f_min);
652  mpp_max_double(nx1*ny1, monotone_data[n].f_max);
653  }
654  }
655 
656  /* adjust the exchange grid cell data to make it monotonic */
657  for(n=0; n<interp[m].nxgrid; n++) {
658  i1 = interp[m].i_in [n];
659  j1 = interp[m].j_in [n];
660  tile = interp[m].t_in [n];
661  n1 = j1*nx1+i1;
662  n2 = (j1+1)*(nx1+2)+i1+1;
663  f_bar = field_in[tile].data[n2];
664  if(xdata[n] == missing) continue;
665 
666  if( monotone_data[tile].f_max[n1] > monotone_data[tile].f_bar_max[n1] ) {
667  /* z1l: Due to truncation error, we might get xdata[n] > f_bar_max[n1]. So
668  we allow some tolerance. What is the suitable tolerance? */
669  xdata[n] = f_bar + ((xdata[n]-f_bar)/(monotone_data[tile].f_max[n1]-f_bar))
670  * (monotone_data[tile].f_bar_max[n1]-f_bar);
671  if( xdata[n] > monotone_data[tile].f_bar_max[n1]) {
672  if(xdata[n] - monotone_data[tile].f_bar_max[n1] < TOLERANCE ) xdata[n] = monotone_data[tile].f_bar_max[n1];
673  if( xdata[n] > monotone_data[tile].f_bar_max[n1]) {
674  printf(" n = %d, n1 = %d, xdata = %f, f_bar_max=%f\n", n, n1, xdata[n], monotone_data[tile].f_bar_max[n1]);
675  mpp_error(" xdata is greater than f_bar_max ");
676  }
677  }
678  }
679  else if( monotone_data[tile].f_min[n1] < monotone_data[tile].f_bar_min[n1] ) {
680  /* z1l: Due to truncation error, we might get xdata[n] < f_bar_min[n1]. So
681  we allow some tolerance. What is the suitable tolerance? */
682  xdata[n] = f_bar + ((xdata[n]-f_bar)/(monotone_data[tile].f_min[n1]-f_bar)) * (monotone_data[tile].f_bar_min[n1]-f_bar);
683  if( xdata[n] < monotone_data[tile].f_bar_min[n1]) {
684  if(monotone_data[tile].f_bar_min[n1] - xdata[n]< TOLERANCE ) xdata[n] = monotone_data[tile].f_bar_min[n1];
685  if( xdata[n] < monotone_data[tile].f_bar_min[n1]) {
686  printf(" n = %d, n1 = %d, xdata = %f, f_bar_min=%f\n", n, n1, xdata[n], monotone_data[tile].f_bar_min[n1]);
687  mpp_error(" xdata is less than f_bar_min ");
688  }
689  }
690  }
691  }
692  for(n=0; n<ntiles_in; n++) {
693  free(monotone_data[n].f_bar_max);
694  free(monotone_data[n].f_bar_min);
695  free(monotone_data[n].f_max);
696  free(monotone_data[n].f_min);
697  }
698 
699  /* remap onto destination grid */
700  for(n=0; n<interp[m].nxgrid; n++) {
701  i2 = interp[m].i_out[n];
702  j2 = interp[m].j_out[n];
703  i1 = interp[m].i_in [n];
704  j1 = interp[m].j_in [n];
705  tile = interp[m].t_in [n];
706  area = interp[m].area [n];
707  if(xdata[n] == missing) continue;
708  if(weight_exist) area *= grid_in[tile].weight[j1*nx1+i1];
709  n1 = j1*nx1+i1;
710  n0 = j2*nx2+i2;
711  if( cell_measures )
712  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
713  else if( cell_methods == CELL_METHODS_SUM )
714  area /= grid_in[tile].cell_area[n1];
715  field_out[m].data[n0] += xdata[n]*area;
716  out_area[n0] += area;
717  }
718  free(xdata);
719  }
720  else {
721  if(has_missing) {
722  for(n=0; n<interp[m].nxgrid; n++) {
723  i2 = interp[m].i_out[n];
724  j2 = interp[m].j_out[n];
725  i1 = interp[m].i_in [n];
726  j1 = interp[m].j_in [n];
727  di = interp[m].di_in[n];
728  dj = interp[m].dj_in[n];
729  tile = interp[m].t_in [n];
730  area = interp[m].area [n];
731  nx1 = grid_in[tile].nx;
732  ny1 = grid_in[tile].ny;
733 
734  if(weight_exist) area *= grid_in[tile].weight[j1*nx1+i1];
735  n2 = (j1+1)*(nx1+2)+i1+1;
736  n0 = j2*nx2+i2;
737 
738  if( field_in[tile].data[n2] != missing ) {
739  n1 = j1*nx1+i1;
740  n0 = j2*nx2+i2;
741  if( cell_measures )
742  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
743  else if( cell_methods == CELL_METHODS_SUM )
744  area /= grid_in[tile].cell_area[n1];
745  if(field_in[tile].grad_mask[n1]) { /* use zero gradient */
746  field_out[m].data[n0] += field_in[tile].data[n2]*area;
747  }
748  else {
749  field_out[m].data[n0] += (field_in[tile].data[n2]+field_in[tile].grad_x[n1]*di
750  +field_in[tile].grad_y[n1]*dj)*area;
751  }
752  out_area[n0] += area;
753  }
754  }
755  }
756  else {
757  for(n=0; n<interp[m].nxgrid; n++) {
758  i2 = interp[m].i_out[n];
759  j2 = interp[m].j_out[n];
760  i1 = interp[m].i_in [n];
761  j1 = interp[m].j_in [n];
762  di = interp[m].di_in[n];
763  dj = interp[m].dj_in[n];
764  tile = interp[m].t_in [n];
765  area = interp[m].area [n];
766 
767  nx1 = grid_in[tile].nx;
768  ny1 = grid_in[tile].ny;
769  if(weight_exist) area *= grid_in[tile].weight[j1*nx1+i1];
770  for(k=0; k<nz; k++) {
771  n0 = k*nx2*ny2 + j2*nx2+i2;
772  n1 = k*nx1*ny1+j1*nx1+i1;
773  n2 = k*(nx1+2)*(ny1+2)+(j1+1)*(nx1+2)+i1+1;
774  if( cell_measures )
775  area *= (field_in[tile].area[n1]/grid_in[tile].cell_area[n1]);
776  else if( cell_methods == CELL_METHODS_SUM )
777  area /= grid_in[tile].cell_area[n1];
778  field_out[m].data[n0] += (field_in[tile].data[n2]+field_in[tile].grad_x[n1]*di
779  +field_in[tile].grad_y[n1]*dj)*area;
780  out_area[n0] += area;
781  }
782  }
783  }
784  }
785 
786  if(opcode & CHECK_CONSERVE) {
787  for(i=0; i<nx2*ny2*nz; i++) {
788  if(out_area[i] > 0) gsum_out += field_out[m].data[i];
789  }
790  }
791 
792  if( cell_measures || ( !(opcode & TARGET) && !(cell_methods == CELL_METHODS_SUM)) ) {
793  for(i=0; i<nx2*ny2*nz; i++) {
794  if(out_area[i] > 0)
795  field_out[m].data[i] /= out_area[i];
796  else
797  field_out[m].data[i] = missing;
798  }
799  }
800  else if( opcode & TARGET ) {
801  for(i=0; i<nx2*ny2; i++) {
802  if(out_area[i] > 0)
803  for(k=0; k<nz; k++) field_out[m].data[k*nx2*ny2+i] /= grid_out[m].cell_area[i];
804  else
805  for(k=0; k<nz; k++) field_out[m].data[k*nx2*ny2+i] = missing;
806  }
807  }
808  else if ( cell_methods == CELL_METHODS_SUM ) {
809  for(i=0; i<nx2*ny2*nz; i++) {
810  if(out_area[i] == 0)
811  for(k=0; k<nz; k++) field_out[m].data[k*nx2*ny2+i] = missing;
812  }
813  }
814 
815  free(out_area);
816  }
817 
818 
819  /* conservation check if needed */
820  if(opcode & CHECK_CONSERVE) {
821  double gsum_in, dd;
822  gsum_in = 0;
823  for(n=0; n<ntiles_in; n++) {
824 
825  nx1 = grid_in[n].nx;
826  ny1 = grid_in[n].ny;
827 
828 
829  if( cell_measures ) {
830  for(j=0; j<ny1; j++) for(i=0; i<nx1; i++) {
831  dd = field_in[n].data[(j+halo)*(nx1+2*halo)+i+halo];
832  if(dd != missing) gsum_in += dd*field_in[n].area[j*nx1+i];
833  }
834  }
835  else if ( cell_methods == CELL_METHODS_SUM ) {
836  for(j=0; j<ny1; j++) for(i=0; i<nx1; i++) {
837  dd = field_in[n].data[(j+halo)*(nx1+2*halo)+i+halo];
838  if(dd != missing) gsum_in += dd;
839  }
840  }
841  else {
842  for(k=0; k<nz; k++) for(j=0; j<ny1; j++) for(i=0; i<nx1; i++) {
843  dd = field_in[n].data[k*(nx1+2*halo)*(ny1+2*halo)+(j+halo)*(nx1+2*halo)+i+halo];
844  if(dd != missing) gsum_in += dd*grid_in[n].cell_area[j*nx1+i];
845  }
846  }
847  }
848  mpp_sum_double(1, &gsum_out);
849 
850  if(mpp_pe() == mpp_root_pe()) printf("the flux(data*area) sum of %s: input = %g, output = %g, diff = %g. \n",
851  field_in->var[varid].name, gsum_in, gsum_out, gsum_out-gsum_in);
852 
853  }
854

859 /*******************************************************************************
860  void do_vector_conserve_interp( )
861  doing conservative interpolation
862 *******************************************************************************/
863 void do_vector_conserve_interp(Interp_config *interp, int varid, int ntiles_in, const Grid_config *grid_in, int ntiles_out,
864  const Grid_config *grid_out, const Field_config *u_in, const Field_config *v_in,
865  Field_config *u_out, Field_config *v_out, unsigned int opcode)
866 {
867  int nx1, ny1, nx2, ny2, i1, j1, i2, j2, tile, n, m, i;
868  double area, missing, tmp_x, tmp_y;
869  double *out_area;
870 
871  missing = u_in->var[varid].missing;
872  /* first rotate input data */
873  for(n = 0; n < ntiles_in; n++) {
874  if(grid_in[n].rotate) {
875  nx1 = grid_in[n].nx;
876  ny1 = grid_in[n].ny;
877  for(i=0; i<nx1*ny1; i++) {
878  tmp_x = u_in[n].data[i];
879  tmp_y = v_in[n].data[i];
880  if( tmp_x != missing && tmp_y != missing) {
881  u_in[n].data[i] = tmp_x * grid_in[n].cosrot[i] - tmp_y * grid_in[n].sinrot[i];
